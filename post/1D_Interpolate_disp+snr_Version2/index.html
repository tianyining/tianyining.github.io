import os
import glob
import math
import numpy as np
import matplotlib.pyplot as plt
from scipy import interpolate
from pandas import DataFrame
from jupyterthemes import jtplot

jtplot.style(theme="monokai")

input_dir_disp = '../disp_PKU/'
input_dir_snr = '../snr_PKU/'
output_dir = "../disp_snr_Interp/"
'''
 遍历所有的 single_disp_file
'''
for single_disp_file in glob.glob(input_dir_disp + "*.SAC_s_2_DISP.1"):
    #     print(single_disp_file)
    # 读取单一频散文件全部信息 single_disp_info
    fileID = open(single_disp_file, "r")
    single_disp_info = fileID.readlines()
    fileID.close()
    # 读取单一频散文件中的 period_disp ph_V 信息，并存为列表
    period_disp = []  # 作为 x_new
    ph_V = []
    for i in range(0, len(single_disp_info)):
        sgl_line = single_disp_info[i]
        elements = sgl_line.strip("\n").split()
        period_ele = float(elements[2])
        phV_ele = float(elements[4])
        period_disp.append(period_ele)
        ph_V.append(phV_ele)
    '''
    读取单一频散文件对应的信噪比文件
    '''
    prename = os.path.basename(single_disp_file).split('.')[0]
    #     print("prename = ", prename)  # prename 表示台站对 suffix 表示后缀 'SAC_s_2_DISP'
    suffix = ".SAC_s_snr.txt"
    fileID = open(input_dir_snr + prename + suffix)
    single_snr_info = fileID.readlines()
    fileID.close()
    period_snr = []  # x
    snr_s = []  # y
    for i in range(0, len(single_snr_info)):
        sgl_line = single_snr_info[i]
        elements = sgl_line.strip("\n").split()
        period_ele = float(elements[0])
        snr_ele = float(elements[1])
        period_snr.append(period_ele)
        snr_s.append(snr_ele)
    '''
    Interpolate Begin
    '''
    x = period_snr
    y = snr_s
    x_new = period_disp

    kind_list = ["linear", "nearest"]
    for kind in kind_list:
        f = interpolate.interp1d(x, y, kind=kind)
        y_new = f(x_new)
    '''
    将 period_disp   ph_V   snr(y_new)  转化成 ndarray 格式 
    '''
    period_disp_array = np.array(period_disp)[:, np.newaxis]
    ph_V_array = np.array(ph_V)[:, np.newaxis]
    snr_interp_array = np.array(y_new)[:, np.newaxis]
    concatenate_array = np.concatenate(
        (period_disp_array, ph_V_array, snr_interp_array), axis=1)

    single_disp_file = DataFrame(
        concatenate_array,
        columns=['period', 'ph_V', 'SNR'],
    )
    # print(single_disp_file)
    single_disp_file.to_csv(output_dir + prename + ".csv", index=None)
