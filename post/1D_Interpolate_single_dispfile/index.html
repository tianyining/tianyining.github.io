本篇 blog 记录利用 python 对单一频散文件进行插值的示例，实现将 AFTAN 计算出来的某一个频散文件（原始未经任何处理），对其中的 period phase_velocity 相应的两列数据进行插值，并绘图对比插值前后的相速度频散趋势，若插值前后相速度频散趋势无变化，证明插值方法可行，不影响最终数据本身表达。

list2 = [0, 1, 2, 3]

from scipy import interpolate
import numpy
import matplotlib.pyplot as plt
from jupyterthemes import jtplot

jtplot.style(theme='monokai')  # 选择一个绘图主题即可

x = period_list
y = phase_velocity_list

# 定义插值点 new_period_list
new_period_list = []
for new_period in range(4, 29):
    #     print(new_period)
    new_period_list.append(new_period)
print(period_list) 
print(new_period_list)

# x_new = new_period_list
# 进行插值
# 定义函数  x:横坐标列表 y:纵坐标列表 kind:插值方式
# f = interpolate.interp1d(x, y, kind='cubic')
# f = interpolate.interp1d(x, y, kind='slinear')
f = interpolate.interp1d(x, y,kind="linear")
#定义新的横坐标列表
x_new = new_period_list
y_new = list(f(x_new))
# plt.plot(x,y,'r',label='original values')

'''
将插值过后的 period_list & phase_velocity_list 写入到新的文件 
disp_phase_interpolate.csv 中
'''
import numpy as np
from pandas import DataFrame
import pandas as pd

len(x_new) == len(y_new)

period_interpolate = x_new
phase_velocity_interpolate = y_new
# print(period_interpolate)
# print(phase_velocity_interpolate)

period_array = np.array(period_interpolate)[:, np.newaxis]
# print('period_array = \n', period_array)
phase_velocity_array = np.array(phase_velocity_interpolate)[:, np.newaxis]
# print("phase_velocity_interpolate = \n", phase_velocity_array)
print('period_array.shape = ', period_array.shape)
print('phase_velocity_array.shape = ', phase_velocity_array.shape)

# 将 period_array & phase_velocity_array 拼接成一个 array
concatenate_array = np.concatenate((period_array, phase_velocity_array),
                                   axis=1)

print(concatenate_array)
print('concatenate_array.shape = ', concatenate_array.shape)

# 将 concatenate_array 转化成 DataFrame 类型，为最终保存为 csv 格式做准备
disp_data_interpolate = DataFrame(concatenate_array,
                                  columns=["period", "ph_V"])

print('disp_data_interpolate = \n', disp_data_interpolate)
# print(type(disp_data_interpolate))

disp_data_interpolate.to_csv("./disp_phase_interpolate.csv")

'''
将插值后的相速度频散曲线与插值前的相速度频散曲线做对比；
画在同一张图上；
'''

import matplotlib.pyplot as plt
import os
import csv
import pandas as pd
from jupyterthemes import jtplot

jtplot.style('monokai')
'''
提取出 period_list
'''
file = open('./disp_phase.txt', 'r')
period_list = []
for file_info in file.readlines():
    c_array = file_info.split(' ')
    period = c_array[0]
    period_list.append(period)
file.close()
period_list = list(map(float, period_list))
'''
提取出 phase_velocity_list
'''
file = open('./disp_phase.txt', 'r')
phase_velocity_list = []
for file_info in file.readlines():
    c_array = file_info.split(" ")
    phase_velocity = c_array[1]
    #     print(phase_velocity)
    phase_velocity_list.append(float(phase_velocity))
file.close()
phase_velocity_list = list(map(float, phase_velocity_list))
'''
提取出 period_list_interpolate
'''
with open('./disp_phase_interpolate.csv', 'r') as f:
    reader = csv.reader(f)
    column = [row[1] for row in reader]
    #     print(column)
    column_period = column[1:]
    #     print(column_period)
    column_period_float = list(map(float, column_period))
    print('column_period_float = \n', column_period_float)
'''
提取出 phase_velocity_list_interpolate 
'''
with open('./disp_phase_interpolate.csv', 'r') as f:
    reader = csv.reader(f)
    column = [row[2] for row in reader]
    column_phase_velocity = column[1:]
    #     print(column_phase_velocity)
    column_phase_velocity_float = list(map(float, column_phase_velocity))
    print("column_phase_velocity_float = \n", column_phase_velocity_float)
'''
Matplotlib 绘图
'''
x = period_list
y = phase_velocity_list
x_new = column_period_float
y_new = column_phase_velocity_float
plt.title("Compare Analysis")
plt.plot(x, y, label='Before Interpolate', color='b')
plt.plot(x_new, y_new, label='After Interpolate', color='g')

plt.xlim((7.5, 33))
