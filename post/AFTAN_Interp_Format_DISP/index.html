说明

本篇 blog 记录利用 python 实现将 AFTAN 提取出的原始频散文件，利用样条插值，整合计算出的信噪比文件（包括插值部分），合并整理成以台站对命名的频散文件，最终每个频散文件包含四列信息，实现批处理。

* 整数周期(int_period)
* 群速度(groupv_itp)
* 相速度(phasev_itp)
* 信噪比(snr_iterp1d)，

注意

本次重新调用 AFTAN 程序，提取了频散曲线，修改了计算互相关对称分量信噪比的程序，lim_rms/spectral_snr_f_V2 中将第 134 135 行的最短、最长周期进行修改，原本 (5,100)，修改为 (1.0,120.0) ，这样做的目的是为了计算更长周期的信噪比，当提取长周期的频散曲线时（本次 period_min = 2 period_max = 100），在利用整数周期对信噪比进行插值并整合频散文件时，可以保证插值的数据点(int_period)不超出原始数据范围(period_snr)。

否则 python 会报错。

相较于 Version1 的提升

Version1

相较于 Version1 ，Version2 将代码更加简化，首先使用样条插值的方法，得到整数周期对应的群速度、相速度，再利用插值得到的整数周期 int_period 对 snr_s 进行插值，最终整合频散文件。

import os
import glob
import numpy as np
import pandas as pd
from scipy import interpolate

input_disp_dir = "../MM_DISP_FTAN/"
input_snr_dir = "../MM_SNR/"
output_dir = "../Bspline_ITP/"

for sgl_disp in glob.glob(input_disp_dir + "*.SAC_s_2_DISP.1"):
    sgl_disp_name = os.path.basename(sgl_disp)
    filein_disp = sgl_disp
    namescol_disp = [
        "index", "period1", "period2", "groupv", "phasev", "xxx1", "xxx2"
    ]
    dataset = pd.read_table(filein_disp,
                            delim_whitespace=True,
                            names=namescol_disp)
    period_disp = list(dataset.period2 * 1000)
    groupv_disp = list(dataset.groupv)
    phasev_disp = list(dataset.phasev)
    '''
    读取 disp 文件对应的 snr 文件；
    '''
    prename = sgl_disp_name.split(".")[0]
    suffix = ".SAC_s_snr.txt"
    sgl_snr_name = sgl_disp_name.split(".")[0] + suffix

    filein_snr = input_snr_dir + sgl_snr_name
    namescol_snr = ["period_snr", "snr"]
    datatab = pd.read_table(filein_snr,
                            delim_whitespace=True,
                            names=namescol_snr)
    '''
    bspline_ITP 部分，将定义的函数融合进来；
    '''
    # sort by x
    INFO = np.transpose([period_disp, groupv_disp, phasev_disp])
    INFO = sorted(INFO, key=lambda FirstINFO: FirstINFO[0])
    INFO = np.transpose(INFO)
    x, y, z = INFO[0], INFO[1], INFO[2]

    # remove duplicate item 删掉重复周期以及其对应的数值
    x_diff = x[1:] - x[:-1]
    index = np.where(x_diff == 0)[0]

    if index == []:
        pass
    else:
        for ii in index[::-1]:
            x, y, z = np.delete(x, ii), np.delete(y, ii), np.delete(z, ii)
    '''
    Interpolate Begin !!! 
    '''
    interval = 1
    if len(x) <= 3:
        x_itp, y_itp, z_itp = [], [], []
    else:
        x_itp = np.arange(x[0], x[-1] + interval, interval)
        xck = interpolate.splrep(x, y)
        zck = interpolate.splrep(x, z)
        y_itp = interpolate.splev(x_itp, xck)
        z_itp = interpolate.splev(x_itp, zck)
        print(x_itp, y_itp, z_itp)  # 插值后的 周期 群速度 相速度；
    '''
    将插值后的结果，写入新的文件；
    '''
    Period_itp = x_itp
    GroupV_itp = y_itp
    PhaseV_itp = z_itp
    for nn in range(len(Period_itp)):
        tmpper = int(Period_itp[nn])
        if tmpper % 1000 != 0:
            continue
        tmpGroup = GroupV_itp[nn]
        tmpPhase = PhaseV_itp[nn]
    '''
    写入新文件
    '''
    fileID = open(output_dir + prename + "_DISP.txt", "w")
    for nn in range(len(Period_itp)):
        tmpper = int(Period_itp[nn])
        if tmpper % 1000 != 0:
            continue
        tmpGroup = GroupV_itp[nn]
        tmpPhase = PhaseV_itp[nn]

        fileID.write("{:<6d} {:<10.4f} {:<10.4f}\n".format(
            tmpper // 1000, tmpGroup, tmpPhase))
        fileID.flush()

import os
import glob
import numpy as np
import pandas as pd
from scipy import interpolate

input_disp_dir = "../Bspline_ITP/"
input_snr_dir = "../MM_SNR/"
output_dir = "../Merge_DISP_SNR/"

for sgl_disp in glob.glob(input_disp_dir + "*txt"):
    sgl_disp_name = os.path.basename(sgl_disp)
    filein_disp = sgl_disp
    namescol_disp = ["int_period", "groupv_itp", "phasev_itp"]
    dataset = pd.read_table(filein_disp,
                            delim_whitespace=True,
                            names=namescol_disp)
    period_disp = list(dataset.int_period)
    groupv_disp = list(dataset.groupv_itp)
    phasev_disp = list(dataset.phasev_itp)
    '''
    读取 disp 文件对应的 snr 文件；
    '''
    prename = sgl_disp_name.split(".")[0][0:-5]
    suffix = ".SAC_s_snr.txt"
    sgl_snr_name = prename + suffix

    filein_snr = input_snr_dir + sgl_snr_name
    namescol_snr = ["period_snr", "snr"]
    datatab = pd.read_table(filein_snr,
                            delim_whitespace=True,
                            names=namescol_snr)

    period_snr = list(datatab.period_snr)
    snr_s = list(datatab.snr)
    '''
    定义插值函数 f
    '''
    x = period_snr
    y = snr_s
    x_new = period_disp
    f = interpolate.interp1d(x, y, kind="linear")
    print(sgl_disp_name)
    y_new = f(x_new)
    '''
    将 linear 一维线性插值后的结果写入新的频散文件；
    '''
    period_disp_array = np.array(period_disp)[:, np.newaxis]
    groupv_disp_array = np.array(groupv_disp)[:, np.newaxis]
    phasev_disp_array = np.array(phasev_disp)[:, np.newaxis]
    snr_s_array = np.array(y_new)[:, np.newaxis]

    concatenate_array = np.concatenate(
        (period_disp_array, groupv_disp_array, phasev_disp_array, snr_s_array),
        axis=1)
    sgl_merge_disp_file = pd.DataFrame(
        concatenate_array, columns=["period", "groupV", "phaseV", "SNR"])
    sgl_merge_disp_file.to_csv(output_dir + prename + "_Merge_Disp.txt",
                               index=None,
                               sep="\t",
                               header=None)
