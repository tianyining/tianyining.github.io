import os
import glob
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
# import pandas_bokeh
from jupyterthemes import jtplot

jtplot.style(theme="monokai")

input_dir = "../PRED_CSV/"
output_dir = "../Figs/"

filein = input_dir + "COR_BAS_CAY.SAC_PRED"
namescol = ["period", "pred_phv"]
datatab = pd.read_table(filein, delim_whitespace=True, names=namescol)

plt.rcParams['axes.linewidth'] = 1  # 图框宽度
plt.rcParams['figure.dpi'] = 720  # plt.show显示分辨率

plt.title("COR_BAS_CAY.SAC_PRED_Scatterplot")
plt.xlim((0, 110))
plt.ylim((0, 5))
plt.scatter(x=datatab.period, y=datatab.pred_phv)
plt.xlabel(
    "Period(s)",
    fontweight='normal',
    size=14,
)
plt.ylabel("PRED_Phv(km/s)", fontweight='normal', size=14)

# plt.show()
plt.savefig(output_dir + "COR_BAS_CAY.SAC_PRED_Scatterplot.png", dpi=720)
