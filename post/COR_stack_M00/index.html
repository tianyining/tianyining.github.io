import obspy
import os
import glob
from matplotlib import pyplot as plt
from jupyterthemes import jtplot
from obspy import Stream
jtplot.style(theme="monokai")

M00_data = glob.glob("../data_MM/I2/M00/*.SAC")
t0 = obspy.UTCDateTime('2021-10-07T00:00:00')

stream = Stream()

for trace in M00_data:
    tr = obspy.read(trace)
    tr[0].stats['distance'] = tr[0].stats.sac.dist * 1000
    tr[0].stats.starttime = t0
    stream += tr[0]

stream.plot(
    type='section',
    size=(600, 800),
    dpi=200,
    #             offset_max=1300 * 1000,
    fig=None,
    #     offset_max = 1300 * 1000,
    scale=2.0,
    orientation='horizontal',
    reftime=t0 + 5000,
    recordstart=-500,
    recordlength=1000,
    linewidth=0.5,
    grid_linewidth=0,
    alpha=1,
    color='b',
    #     outfile='./figs/COR_M00.png',
    #     show= True,
)

import obspy
import os
import glob
from matplotlib import pyplot as plt
from jupyterthemes import jtplot
from obspy import Stream
jtplot.style(theme="monokai")

M00_data = glob.glob("../data_MM/I2/M00/*.SAC")
t0 = obspy.UTCDateTime('2021-10-07T00:00:00')

stream = Stream()

for trace in M00_data:
    tr = obspy.read(trace)
    tr[0].filter("bandpass",
                 freqmin=1 / 40,
                 freqmax=1 / 5,
                 corners=1,
                 zerophase=True)
    tr[0].stats['distance'] = tr[0].stats.sac.dist * 1000
    tr[0].stats.starttime = t0
    stream += tr[0]

stream.plot(
    type='section',
    size=(600, 800),
    dpi=200,
    #             offset_max=1300 * 1000,
    fig=None,
    #     offset_max = 1300 * 1000,
    scale=2.0,
    orientation='horizontal',
    reftime=t0 + 5000,
    recordstart=-500,
    recordlength=1000,
    linewidth=0.5,
    grid_linewidth=0,
    alpha=1,
    color='b',
#         outfile='./figs/COR_M00_filter.png',
    #     show= True,
)
