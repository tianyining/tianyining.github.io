说明

本篇 blog 用于记录如何使用 DSurfTomo 进行检测板恢复测试，主要包含如何设计棋盘网格的速度异常尺寸，生成棋盘网格模型，以及 Synthetic Test 恢复性测试。本篇 blog 使用的测试数据为 example_EHS 。

Step1_Generate_MOD.true

进行棋盘测试需要四个输入文件：

* DSurfTomo.in 参数控制文件；
* MOD：反演所需的初始速度模型；

#!/usr/bin/env python
import numpy as np
import matplotlib.pyplot as plt
import math

# dont have last layer whitch depth == 52
inpfile = "../InpFiles/depth_average_vs_EHS_modified"
output_dir = "../OutFiles/"
data = np.loadtxt(inpfile)
shape = data.shape
v = {}  # velocity at each depth ,key is depth
a = {}  # amplitude at each depth,key is depth a[x]=0.08*v[x]
nx = 12
ny = 19
nz = 13
anosize = math.pi / 4  # nosize*number/pi=number of anomalous

for d in range(shape[0]):
    depth = int(data[d][0])
    v[depth] = data[d][1]
    a[depth] = 0.08 * v[depth]
    if d == (shape[0] - 1):
        v[depth + 5] = v[depth]
        a[depth + 5] = a[depth]

nmax = np.max([nx, ny])
x = range(1, nmax + 1)
y = range(1, nmax + 1)
z = np.zeros(nz) + 1
xy = np.kron(np.sin(anosize * np.array(y)), np.sin(anosize * np.array(x)))
#xyz = np.kron(np.sin(anosize_z*np.array(z)),xy)
xyz = np.kron(np.array(z), xy)

pxy = xyz.reshape(nz, nmax, nmax)

mod = np.zeros((nz * ny, nx))

for k in range(nz):
    for j in range(ny):
        for i in range(nx):
            mod[(k) * ny + j, i] = v[5 * k] + pxy[k, j, i] * a[5 * k]

np.savetxt(output_dir + 'MOD_EHS.true', mod, fmt='%4.4f')

'''
 计算反演出的 example_EHS 不同深度的平均剪切波速度；
 存为 d 字典形式
'''
import os
import glob
import linecache
import numpy as np 

file = "../InputFiles/DSurfTomo.inMeasure.dat" 
output_dir = "../OutputFiles_for_GMT/"
data = np.loadtxt(file)
shape = data.shape

depth = data[0][2]  # 将 depth 首先设置为 0.0
count = 0
sums = 0
maxs = data[0][3]  # 读取 Vs
mins = data[0][3]
d = {}  # key is depth,value is average velocity

for i in range(shape[0]):
    if data[i][2] == depth:
        count += 1
        sums = sums + data[i][3]
        if maxs < data[i][3]:
            maxs = data[i][3]
        if mins > data[i][3]:
            mins = data[i][3]
    else:
        d[depth] = sums / count
        count = 1
        depth = data[i][2]
        sums = data[i][3]

    if i == shape[0] - 1:
        depth = data[i][2]
        d[depth] = sums / count

# write into disk to save depth average vs
with open(output_dir + "depth_average_vs_EHS", "w") as fileID:
    for key in d.keys():
        depth = key
        depth_vs = d[key]
        fileID.write("{:<.2f}\t{:<.4f}\n".format(depth, depth_vs))

d = {}  # key is depth, value is depth average vs

    line = tmpinfo[line].strip("\n").split()

        fileID.write('{:<s}\t{:<.2f}\t{:<.2f}\t{:<.4f}\n'.format(
            position, a, a_absolute, d[data[i][2]]))

# dont have last layer whitch depth == 52

a = {}  # amplitude at each depth,key is depth a[x]=0.08*v[x]
nx = 12
ny = 19
nz = 13
anosize = math.pi / 2  # nosize*number/pi=number of anomalous

for d in range(shape[0]):
    depth = int(data[d][0])
    v[depth] = data[d][1]
    a[depth] = 0.08 * v[depth]
    if d == (shape[0] - 1):

z = np.zeros(nz) + 1

            mod[(k) * ny + j, i] = v[5 * k] + pxy[k, j, i] * a[5 * k]
