import os
import glob
import numpy as np
import pandas as pd

output_dir = "../Step2_Sort_StudyRegion_mod_Depthly/Shen_Surf_dat/"

with open(output_dir + "Shen_Surf_Model.dat", "w") as fp:
    for mod in glob.glob(input_dir + "*.mod"):
        mod_name = os.path.basename(mod)
        lon = float(mod_name.split("_")[0])
        lat = float(mod_name.split("_")[1][:-4])

        fileID = open(mod, "r")
        tmpinfos = fileID.readlines()
        fileID.close()

        for i in range(len(tmpinfos)):
            line = tmpinfos[i]

            fp.write("{:<}\t{:<}\t{:<}".format(lon, lat, line))

'''

import os
import glob
import numpy as np
import pandas as pd

input_file = "../Yiming_Myanmar_Vs_model/Vs_model_TP.dat"
output_dir = "../Step3_Myanmar_Yiming_mod/Named_grid/"

test_data = np.loadtxt(input_file)

for i in range(len(test_data)):
    line = test_data[i]
    lon = line[0]
    lat = line[1]
    depth = line[3]
    vs = line[2]

    with open(output_dir + str(lon) + "_" + str(lat), 'a') as fileID:
        fileID.write("{:<5.1f}\t{:<10.4f}\n".format(
            depth, vs))

'''

import os
import glob
import numpy as np
import pandas as pd

Yiming_input = "../Step3_Myanmar_Yiming_mod/Named_Depth/"
Shen_input = "../Step2_Sort_StudyRegion_Shen_mod_Depthly/Shen_mod_Depthly/"
output_dir = "../Step4_Merge_mod_WholeRegion/Whole_Region_mod_depthly/"

for depth in glob.glob(Yiming_input + "*"):
    depth_name = os.path.basename(depth)
    #     print("depth1:",depth_name)
    fileID = open(Yiming_input + depth_name, "r")
    tmpinfo_Yiming = fileID.readlines()
    fileID.close()

    fileID = open(Shen_input + depth_name, "r")
    tmpinfo_Shen = fileID.readlines()
    fileID.close()

    tmpinfo = tmpinfo_Yiming + tmpinfo_Shen
    #     print('depth2:',depth_name)
    with open(output_dir + depth_name, "w") as fp:
        fp.writelines(tmpinfo)
        fp.close()

#!/bin/zsh 

# Date:2022-06-10  
# Author:Yining Tian 
# Function : 绘制 10km 深度 Shen 2016 的 Vs 反演结果；

output_path="../Figs/"

# GMT Settings 
gmt set MAP_FRAME_TYPE Plain 
gmt set MAP_FRAME_PEN 1.0p 
gmt set FONT Times-Roman # 同时设置所有 FONT 类参数，
gmt set FONT_ANNOT_PRIMARY 6p,Times-Roman # 设置一级标注的字体；
gmt set FONT_TITLE 15p,Times-Roman # 图上方标题的字体；  
gmt set MAP_TICK_LENGTH_PRIMARY 1.5p/1p # 一级刻度的主刻度和次刻度的长度 [5p/2.5p]
gmt set MAP_TICK_PEN_PRIMARY 0.5p,black # 一级刻度的画笔属性；
gmt set MAP_TITLE_OFFSET 0p # 图标题的底部与轴标注（或轴标签）的顶部之间的距离 [14p]
gmt set MAP_GRID_PEN_PRIMARY thicker,black # 一级网格的线条属性；
gmt set COLOR_NAN 255/255/255 # 数值 Z 为空的时候，所用的颜色值；

fig_name="Shen_Vs_10"
fig=$output_path$fig_name 

gmt begin $fig png 
    gmt subplot begin 2x4 -Fs6c -B -BWeSn  
    J_tomo="-JM2i"
    R_tomo="-R92.5/105.5/19.0/29.0" # 比实际 tomo 区域大； 
    gmt subplot set 0 
    input_file="../Step2_Sort_StudyRegion_mod_Depthly/Shen_mod_Depthly/10.0"
    RMAP=`cat $input_file | gmt info -C | awk '{print "-R"$1"/"$2"/"$3"/"$4}'`
    echo $RMAP 
    gmt basemap $R_tomo $J_tomo -Ba4f1 -BWSen 
    cat $input_file | awk '{print $1,$2,$4}' > Vs_10.dat 

    gmt xyz2grd Vs_10.dat $R_tomo -I0.5/0.5 -GVs_10.grd 
    gmt grdsample Vs_10.grd -GVs_10.grd  -I0.1/0.1 
    gmt grd2cpt Vs_10.grd  -Cseis -Z -H -D > Vs_10.cpt 
    gmt grdimage Vs_10.grd -CVs_10.cpt $R_tomo $J_tomo -E720

    gmt coast -EFR,IT+gP300/8 -Di -N1/0.015c,-  $R_tomo $J_tomo 
    gmt set FONT_LABEL +8p,Times-Roman,black  
    gmt set FONT_ANNOT_PRIMARY 6p,Times-Roman,black # 边框经纬度注释字体；
    gmt colorbar -CVs_10.cpt  $R_tomo $J_tomo -DJBC+w4c/0.15c+e+mc  -Bxa0.1f0.1+l"Vs (km/s)"  -By

gmt legend $R_tomo $J_tomo  -DjRB+w1.0c+o0.05c/0.05c -F+gwhite -t20 << EOF
H 8p,Times-Roman 10km 
EOF

    gmt subplot end 
gmt end show 

rm gmt* *.cpt *.grd *.dat

#!/bin/zsh 
# Date:2022-06-15 
# Author:Yining Tian 
# Function : 绘制 12km 深度 Yiming Bai 在 Myanmar 以及 Shen 在云南的 Vs 反演结果；

output_path="../Figs/"
Myanmar_mod="../Step3_Myanmar_Yiming_mod/Named_Depth/12.0"
Shen_mod="../Step2_Sort_StudyRegion_Shen_mod_Depthly/Shen_mod_Depthly/12.0"
Whole_Region_mod="../Step5_WholeRegion_Interp_grid/Depth/12_WholeRegion_Grid.txt"

# GMT Settings 
gmt set MAP_FRAME_TYPE Plain 
gmt set MAP_FRAME_PEN 1.0p 
gmt set FONT Times-Roman # 同时设置所有 FONT 类参数，
gmt set FONT_ANNOT_PRIMARY 6p,Times-Roman # 设置一级标注的字体；
gmt set FONT_TITLE 15p,Times-Roman # 图上方标题的字体；  
gmt set MAP_TICK_LENGTH_PRIMARY 1.5p/1p # 一级刻度的主刻度和次刻度的长度 [5p/2.5p]
gmt set MAP_TICK_PEN_PRIMARY 0.5p,black # 一级刻度的画笔属性；
gmt set MAP_TITLE_OFFSET 0p # 图标题的底部与轴标注（或轴标签）的顶部之间的距离 [14p]
gmt set MAP_GRID_PEN_PRIMARY thicker,black # 一级网格的线条属性；
gmt set COLOR_NAN 255/255/255 # 数值 Z 为空的时候，所用的颜色值；

fig_name="WholeRegion_Interp_Vs_12"
fig=$output_path$fig_name 

gmt begin $fig png 
    gmt subplot begin 2x4 -Fs6c -B -BWeSn  
    J_tomo="-JM2i"
    R_tomo="-R91.5/106.5/19.25/29.5" # 比实际 tomo 区域大； 
    gmt subplot set 0 

    # Myanmar_RMAP=`cat $Myanmar_mod | gmt info -C | awk '{print "-R"$1"/"$2"/"$3"/"$4}'`
    # echo $Myanmar_RMAP 
    gmt basemap $R_tomo $J_tomo -Ba4f1 -BWSen 

    # cat $Myanmar_mod | awk '{print $1,$2,$4}' > Myanmar_12.dat 
    # cat $Shen_mod | awk '{print $1,$2,$4}' > Shen_12.dat 
    cat $Whole_Region_mod | awk '{print $1,$2,$3}' > WholeRegion_Interp_12.dat 
    gmt xyz2grd WholeRegion_Interp_12.dat $R_tomo $J_tomo -I0.25/0.25 -GWholeRegion_Interp.grd 
    gmt grdsample WholeRegion_Interp.grd -GWholeRegion_Interp.grd  -I0.05/0.05
    gmt makecpt -Cseis -T2.2/3.63/0.01 -H -D -Z > tmp.cpt 
    gmt grdimage WholeRegion_Interp.grd -Ctmp.cpt $R_tomo $J_tomo -E720

    cat $Myanmar_mod | awk '{print $1,$2}' | gmt plot -Sc0.02c -Gdarkgray $R_tomo $J_tomo
    cat $Shen_mod | awk '{print $1,$2}' | gmt plot -Sc0.02c -Gred $R_tomo $J_tomo

    gmt coast -EFR,IT+gP300/8 -Di -N1/0.015c,-  $R_tomo $J_tomo 
    gmt set FONT_LABEL +8p,Times-Roman,black  
    gmt set FONT_ANNOT_PRIMARY 6p,Times-Roman,black # 边框经纬度注释字体；
    gmt colorbar -Ctmp.cpt  $R_tomo $J_tomo -DJBC+w4c/0.15c+e+mc  -Bxa0.2f0.1+l"Vs (km/s)"  -By

gmt legend $R_tomo $J_tomo  -DjRB+w0.9c+o0.05c/0.05c -F+gwhite -t20 << EOF
H 6p,Times-Roman 12km 
EOF

    gmt subplot end 
gmt end show 

rm gmt* *.cpt *.grd *.dat
