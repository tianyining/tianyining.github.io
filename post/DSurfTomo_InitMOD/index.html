说明

本篇 blog 用于记录如何根据已知的初始速度模型（一维或三维），生层 DSurfTomo 反演所需的初始速度模型 MOD ，1d_MOD 是根据 Yiming Bai 手动挑选的 CDisp 通过计算生成的一维初始速度模型构建的，3d_MOD 是根据 Yiming Bai 文章联合反演得到的区域速度结构扩充构建的。

import os
import glob
import numpy as np
import pandas as pd

input_grid_dir = "../Yiming_TP/Named_grid/" 
input_depth_dir = "../Yiming_TP/Named_Depth/" 
output_dir = "../MOD_Test_Final/MOD_Depth/"  

for depth in glob.glob(input_depth_dir + "*"):
    depth = float(os.path.basename(depth))  

    Lon_Depth = []

    for lon in lon_grid:
        tmp = []
        tmp_name = []
        tmp_name_new = [] 
        for file in glob.glob(input_grid_dir + "*"):
            file_name = os.path.basename(file)
            file_lon = file_name.split("_")[0]
            if file_lon == lon: 
                tmp_name.append(file_name)

        # 将某一固定 lon 的所有纬度格点，由低到高排列
        tmp_lat = []
        for file in tmp_name:
            file_name = os.path.basename(file)
            lat = float(file_name.split("_")[1])
            tmp_lat.append(lat)
        tmp_lat = sorted(tmp_lat) 

        for i in tmp_lat: 
            tmp_new = lon + "_" + str(i)
            tmp_name_new.append(tmp_new)
        for sgl_file in tmp_name_new: 
            namescol = ['depth', 'vs', 'vp', 'rho']

            data = pd.read_csv(input_grid_dir + sgl_file, names=namescol, sep="\s+")

            tmp.append(tmp_vs)

        # 将每一深度的 MOD 格式写入文件 MOD_depth
        Lon_Depth.append(tmp)
        fileID = open(output_dir + "MOD_" + str(depth), "w")
        str1 = "  "
        for lon_info in Lon_Depth:

            fileID.write("\n") 
        fileID.close()
