# 对 AFTAN 提取出的频散文件中的观测周期进行插值：
import os
import glob
import numpy as np
from scipy import interpolate

input_disp = "../YN_SAC_s_2_DISP.1/"
output_dir = "../Step1_YN_DISP_Interp/"
fmt = "{:<f}\t{:<15.6f}\t{:<15.6f}\n"
for disp in glob.glob(input_disp + "*.SAC_s_2_DISP.1"):
    disp_name = os.path.basename(disp)

    fileID = open(disp, "r")
    tmp_info = fileID.readlines()
    fileID.close()

    observe_period = []
    Group_V = []
    Phase_V = []

    for i in range(0, len(tmp_info)):
        sgl_line = tmp_info[i].strip()
        period = float(sgl_line.split()[2])
        group_v = float(sgl_line.split()[3])
        phase_v = float(sgl_line.split()[4])

        observe_period.append(period)
        Group_V.append(group_v)
        Phase_V.append(phase_v)
    '''
    建立 int_period_list 
    '''
    period_list = []
    for item in observe_period[0:-1]:
        item_int = int(item)
        #     print(item_int)
        period_list.append(item_int)
    period_list.append(int(observe_period[-1]))

    period_set = list(set(period_list))
    '''
    Interpolate Begin !!! 
    '''
    x = observe_period
    y = Group_V
    z = Phase_V

    # 定义插值函数 f
    f_groupv = interpolate.interp1d(x,
                                    y,
                                    kind="linear",
                                    fill_value="extrapolate")
    f_phasev = interpolate.interp1d(x,
                                    z,
                                    kind="linear",
                                    fill_value="extrapolate")

    x_new = period_set
    int_Group_V = f_groupv(x_new)
    int_Phase_V = f_phasev(x_new)
    '''
    将一维线性插值后的结果，重新保存为新的频散文件；
    '''
    with open(output_dir + "interp_" + disp_name, "w") as fileID:
        for n in range(0, len(x_new)):
            int_period = x_new[n]
            interp_groupv = int_Group_V[n]
            interp_phasev = int_Phase_V[n]

            fileID.write(fmt.format(int_period, interp_groupv, interp_phasev))

'''
 Right Version 单纯统计射线路径，不剔除3倍标准差以外的数据点；
'''
import os
import math
import numpy as np

input_dir = "../Statistics_Info/"
output_dir = "../Statistics_Info/"
data = np.loadtxt(input_dir + "disper_curves_compute")
shape = data.shape  # (193610, 2)
'''
establish dict
'''
dict_disp = {}
d = {}  # the value of d is standard deviation 标准差
a = {}  # the value of a is average 每个周期对应的相速度平均值；
s = {}  # sum velocity of each period 每个周期的相速度和
n = {}  # get the number of each period 每个周期对应有多少个相速度值
ss = {}  # sum of square(real phV - average)
'''
 已求取出： a-每个周期对应的相速度平均值；
          s-每个周期对应的相速度和;
          n-每个周期上分别有多少个相速度测量值；
'''
for i in range(shape[0]):
    period = int(data[i][0])
    if period not in s.keys():
        s[period] = data[i][2]
        n[period] = 1
    else:
        s[period] = s[period] + data[i][2]
        n[period] = n[period] + 1
    if i == shape[0] - 1:
        for item in s.keys():
            a[item] = s[item] / n[item]
'''
 index j 求取 ss-每个周期对应的平方和（测量值-周期平均值）的平方和 
            d-计算得到每个周期对应的标准差 
'''
for j in range(shape[0]):
    period = int(data[j][0])
    if period not in ss.keys():
        ss[period] = math.pow((data[j][2] - a[period]), 2)
    else:
        ss[period] = math.pow((data[j][2] - a[period]), 2) + ss[period]
    if j == shape[0] - 1:
        for item in ss.keys():
            d[item] = math.sqrt(ss[item] / n[item])
'''
 将计算结果写入新的频散文件，用于绘图和筛选
 each-period_average_std 三列分别代表：周期 - 周期对应的平均相速度 - 标准差
'''
with open(output_dir + "each-period_average_std", "w") as fp:
    for p in ss.keys():
        fp.write("{:<d} {:<.4f} {:<4f}\n".format(p, a[p], d[p]))

# 某些特定周期对应的相速度取值范围
with open(output_dir + "select-period_phvplus", "w") as fp:
    for item in range(5, 100, 5):
        print(item)
        fp.write("> \n")
        fp.write("{:<d} {:<.6f}\n".format(item, a[item] - 3 * d[item]))
        fp.write("{:<d} {:<.6f}\n".format(item, a[item] + 3 * d[item]))
'''
 计算自定义周期的射线路径
'''
t = 0
with open(output_dir + "number_ray_gmt_ori", "w") as fp:

    for item in range(3, 100):
        fp.write("> \n")
        fp.write("{:<.1f} {:<d}\n".format(item - 0.5, n[item]))
        fp.write("{:<.1f} {:<d}\n".format(item + 0.5, n[item]))
        fp.write("{:<.1f} {:<d}\n".format(item + 0.5, 0))
        fp.write("{:<.1f} {:<d}\n".format(item - 0.5, 0))

s = {}  # sum of phase velocity of each period
n = {}  # number of phase velocity of each period
ss = {}  # sum of square

for i in range(shape[0]):
    period = int(data[i][0])
    if period not in s.keys():
        s[period] = data[i][2]
        n[period] = 1
    else:
        s[period] = s[period] + data[i][2]
        n[period] = n[period] + 1
    if i == shape[0] - 1:
        for item in s.keys():
            a[item] = s[item] / n[item]
# index j 求取 ss 每个周期对应的平方和（测量值 - 周期平均值）的平方和 d 是计算得到的不同周期的标准差；
for j in range(shape[0]):
    period = int(data[j][0])
    if period not in ss.keys(): 
        ss[period] = math.pow((data[j][2] - a[period]), 2)
    else:
        ss[period] = math.pow((data[j][2] - a[period]), 2) + ss[period]
    if j == shape[0] - 1:
        for item in ss.keys():
            d[item] = math.sqrt(ss[item] / n[item])

# 首先读取对应台站对的频散文件 & 互相关对称分量计算出的信噪比文件
import os
import glob
import pandas as pd
from scipy import interpolate

input_disp_dir = "../Step2_1.5Wavelength/YN_Disp_Tmax/"
input_snr_dir = "../snr_s_YN/"
output_dir = "../Step3_YN_DISP_SNR_Merge/"

for disp in glob.glob(input_disp_dir + "*"):
    disp_name = os.path.basename(disp)[:-10]  # the same as ccf name
    namescol = ["period", "group_v", "phase_v"]

    dataset = pd.read_csv(
        disp,
        names=namescol,
        delim_whitespace=True,
    )

    int_period = dataset["period"]  # 作为 x_new
    group_v = dataset["group_v"]
    phase_v = dataset["phase_v"]
    '''
    读取对应台站对的 snr_s.txt 文件
    '''
    snr_name = disp_name + "_snr.txt"

    namescol_snr = ["period", "snr_s"]
    dataset_snr = pd.read_csv(
        input_snr_dir + snr_name,
        names=namescol_snr,
        delim_whitespace=True,
    )

    snr_period = dataset_snr["period"]  # x
    snr_s = dataset_snr["snr_s"]  # y

    # Interpolate Begin !!!
    x = snr_period
    y = snr_s
    # 定义插值函数 f_snr
    f_snr = interpolate.interp1d(x, y, fill_value="extrapolate", kind="linear")

    x_new = int_period
    interp_snr = f_snr(x_new)
    '''
    将 snr_s 一维线性插值后的结果与 disp 整合，生成带有信噪比信息的频散文件；
    '''
    with open(output_dir + "Merge_Disp_SNR_" + disp_name, "w") as fileID:
        for n in range(0, len(int_period)):
            p = int_period[n]
            G_v = group_v[n]
            Pha_v = phase_v[n]
            SNR_s = interp_snr[n]

            fileID.write("{:<.1f}\t{:<15.6f}\t{:<15.6f}\t{:<15.6f}\n".format(
                p, G_v, Pha_v, SNR_s))

Establish dict to save statistical info of Myanmar

a = {}  # the value of average phase velocity of each period

n = {}  # the total number of each period
ss = {}  # sum of square (real phaseV - average_phaseV)
PhaseV_min = {}
PhaseV_max = {}

# index i
for i in range(shape[0]):
    period = int(data[i][0])

s = dict(sorted(s.items(), key=lambda x: x[0]))
n = dict(sorted(n.items(), key=lambda x: x[0]))
