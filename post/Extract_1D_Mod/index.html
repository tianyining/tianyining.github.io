'''
 筛选出 lon:97 ~ 104 
       lat:21 ~ 28 范围内的格点文件；
'''
import os
import glob
import numpy as np

input_dir = "../China_2015_Vs_v1.0/"
output_dir = "../YunNan_Region/"

lon1, lon2 = 97, 104
lat1, lat2 = 21, 28

for file in glob.glob(input_dir + "*.mod"):
    file_name = os.path.basename(file)
    tmp_lon = float(file_name.split("_")[0])
    tmp_lat = float(file_name.split("_")[1][:-4])  
    if lon1 <= tmp_lon <= lon2:
        if lat1 <= tmp_lat <= lat2:
            print(file_name)
            os.system("cp " + file + " " + output_dir)

import os
import linecache
import numpy as np

# lon lat depth Vs
files = "../Step3_Calculate_Ave_Depth_Vs_YN_Shen/Depth_YunNan_Shen_DSurfTomo_Format"

data = np.loadtxt(files)
shape = data.shape
depth = data[0][2]  # 将 depth 首先设置为 0.0
count = 0
sums = 0
maxs = data[0][3]  # 读取 Vs
mins = data[0][3]
d = {}  # key is depth,value is average velocity

for i in range(shape[0]):
    if data[i][2] == depth:
        count += 1
        sums = sums + data[i][3]
        if maxs < data[i][3]:
            maxs = data[i][3]
        if mins > data[i][3]:
            mins = data[i][3]
    else:
        d[depth] = sums / count
        count = 1
        depth = data[i][2]
        sums = data[i][3]

    if i == shape[0] - 1:
        depth = data[i][2]
        d[depth] = sums / count

#
with open('../Step3_Calculate_Ave_Depth_Vs_YN_Shen/Ave_Shen_YN',
          "w") as fileID:
    for depth in d.keys():
        depth_ave = d[depth]
        fileID.write("{:<}\t{:<.4f}\n".format(depth, depth_ave))

import os
import numpy as np
import pandas

input_file = "../Step3_Calculate_Ave_Depth_Vs_YN_Shen/Ave_Shen_YN"
output_file = "../Step3_Calculate_Ave_Depth_Vs_YN_Shen/Ave_Shen_YN_ladder"

fileID = open(input_file, "r")
tmpinfos = fileID.readlines()
fileID.close()

with open(output_file, "w") as fileID:
    for i in range(len(tmpinfos)):
        line = tmpinfos[i].strip().split()
        depth = float(line[0])
        vs = float(line[1])
        if i < len(tmpinfos) - 1:
            line_ladder = tmpinfos[i + 1].strip().split()
            depth_ladder, vs_ladder = float(line_ladder[0]), float(
                line_ladder[1])
            fileID.write("{:<}\t{:<}\n".format(depth, vs))
            fileID.write("{:<}\t{:<}\n".format(depth,vs_ladder))

'''
 以 5km 为间隔，计算得到 Ave_Vs_ladder 文件 0km 5km ...
'''
import os
import numpy as np
import pandas

input_file = "../Step3_Calculate_Ave_Depth_Vs_YN_Shen/Ave_Shen_YN"
output_file = "../Step3_Calculate_Ave_Depth_Vs_YN_Shen/Appoint_Depth_Ave_Shen_YN"

fileID = open(input_file, "r")
tmpinfos = fileID.readlines() 
fileID.close()

depth_list = []
for d in range(0, 200, 5):
    depth_list.append(d)

with open(output_file, "w") as fileID:
    for i in range(len(tmpinfos) - 10):
        line = tmpinfos[i]
        depth = float(line.strip().split()[0])
        vs = float(line.strip().split()[1])
        tmp_line = tmpinfos[i + 10]
        tmp_vs = float(tmp_line.strip().split()[1])
        if depth in depth_list:
            fileID.write("{:<}\t{:<}\n".format(depth, vs))
            fileID.write("{:<}\t{:<}\n".format(depth, tmp_vs))

#!/bin/zsh 

# Date:2022-05-25 
# Author:Yining Tian 

YN_Shen_Model="../Step3_Calculate_Ave_Depth_Vs_YN_Shen/Ave_Shen_YN"
fig="../YN_Shen_Model_Smooth"
R="-R2.5/5/0/100"
J="-JX1.5i/-2i"

## GMT Settings 
gmt set MAP_FRAME_WIDTH 3p  # width of fancy frame 边框宽度；
gmt set FONT_ANNOT_PRIMARY 8p,Times-Roman,black # 边框经纬度注释字体；
gmt set MAP_TICK_LENGTH_PRIMARY 0.025c 
gmt set FONT_LABEL=10p # 边框注释 Depth Vs 字体大小； 

gmt begin $fig png 
    gmt basemap $R $J  -Bya20f5+l"Depth (km)" -Bxa0.5f0.1+l"Vs (km/s)" -BWNes 
    cat $YN_Shen_Model | awk '{print $2,$1}' | gmt plot $R $J -W0.7p,cyan

## plot legend 
C1="blue" # End mod 
C2="cyan" # Start mod 
gmt set FONT_ANNOT_PRIMARY 8.0p,Times-Roman,black # the height of the symbols should be the same as the label text 
gmt legend -JX8/6.3 $R -DjBL+w1.0i/0.4i+l1.5+o0.05i/0.05i -C0.1i/0.05i -F+gwhite -t30  >> $fig << EOF
S 0.05i -        0.20i $C2              1.0p,$C2               0.27i  Shen model 
EOF

gmt end show 

rm -f *.conf
