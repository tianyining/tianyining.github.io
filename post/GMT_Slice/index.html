#!/bin/bash 
# Origin_Version : Xingli Fan gmt5.4.2
# Modified by : Yining Tian  gmt6
# Date:2022-01-12 

inp3D="../InpFiles/DSurfTomo.inMeasure.dat"
test="../Figs/slice"
lo1=124
lo2=131.3
la1=40
la2=44.5
R="-R"$lo1"/"$lo2"/"$la1"/"$la2
J="-JM2i" 
depth1="4"
# I="0.25""/""0.25"
Faults="../InpFiles/Faults/"
line="../InpFiles/line/"

gmt begin $test png
    gmt set MAP_TICK_LENGTH 2p MAP_ANNOT_OFFSET 2p MAP_FRAME_PEN 0.5p
    gmt set FORMAT_DATE_MAP o FORMAT_TIME_PRIMARY_MAP Character FONT_ANNOT_PRIMARY +8p,Helvetica,black  # colorbar 的刻度，以及边框刻度的字体大小
    # gmt set FONT_LABEL +8p,Helvetica,red  # FONT_LABEL 控制 colorbar annotation 的字体大小 
    gmt set FONT_LABEL +8p,Times-Roman,red 

    # 16p,Helvetica,black
    # 绘制边框 
    gmt basemap $R $J -Bya1f0.5 -Bxa2f1 -BWseN 

    # make seismic tomography cpt 
    # gmt makecpt -Chaxby -T2.4/4.2/0.9 -H > velocity.cpt 
    # gmt makecpt -Cred,white,blue -T2.9/3.5/0.1 -H -Z  > velocity.cpt 
    gmt makecpt -Cred,white,blue -T2.4/4.2/0.9 -H -Z > velocity.cpt  # 加入 -Z 效果完全不一样；
    # make grey cpt 
    gmt makecpt -Cgray -T-7000/7000/1000 -H > g.cpt

    # 绘制地形起伏 
    gmt grdimage @earth_relief_15s $R $J -Cg.cpt  -I+d #+d for default gradient for shadow

gmt clip $R -JM2i -W0.5p,black,-  << EOF
    
    128.174 41.42
    126.512 41.3621
    124.4 40.1
    124.0 40.1
    124.0 44.5
    131.3 44.5
    131.3 42.7
    130.47 42.6
    129.56 42.3

EOF

## 开始绘制 4km 的切片，包含三列 首先挑选出 4km 深度处的反演结果；
awk '{if($3==depth1) print $1,$2,$4}' depth1="4.0" $inp3D > tmp.dat # 生成的 tmp.dat 包含三列：lon lat velocity_S 
gmt xyz2grd tmp.dat  -R124/131.5/40/44.5 -I0.25/0.25 -Gtmp.grd  
gmt grdimage tmp.grd  -Cvelocity.cpt $J $R -E720 -Q 
# stop clip 
gmt clip -C 

# 绘制海岸线
gmt coast $R $J -N1/0.8p -W0.5p -A0/0/1 -Dh 

## 绘制断裂带 & 火山 使用 -S 选项，则仅在数据点所在位置绘制符号
gmt plot $Faults"Tanlu.xy" $R $J -W0.5p,black 
gmt plot $Faults"suture.xy" $R $J -W1p,gray40,- 
gmt plot $Faults"volcanoes.txt" $R $J -Skvolcano/8p -Gblack  

## 绘制 linea & lineb
gmt plot ${line}"linea" $R $J -W0.5p,brown  
gmt plot ${line}"lineb" $R $J -W0.5p,brown 

## 添加 text 统一规定所有文本的字号为 10p(表示字体大小) 4号字体(表示不同字体) -a-10 表示顺时针旋转10° 
echo 127.0 43.5 suture | gmt text $R $J -F+a-10+f10p,4,black 
echo 126.8 43.1 skc | gmt text $R $J -F+a-0.1+f10p,4,red 
gmt text ${Faults}"volcanoes.txt" $R $J -F+f5p,37,red -D0.05i/-0.12i 
echo 126 42.4432 A | gmt text $R $J -F+f12p,4,orange  # 绘制字母 A 
echo 128.70 41.8717 A\' | gmt text $R $J -F+f12p,4,orange  # 绘制字母 A' 
echo 127.5 41 B | gmt text $R $J -F+f12p,4,orange # 绘制字母 B 
echo 128.5 44.2 B\' | gmt text $R $J -F+f12p,4,orange # 绘制字母 B' 

# Right colorbar type 
# gmt colorbar -Cvelocity.cpt -DJBC+e+mc -Bx0.5+l"S Velocity(km/s)" -By+lm  # colorbar1 
# gmt colorbar -Cvelocity.cpt -DJBC+e+mc  -Ba0.4f0.2+l"S Velocity(km/s)"  # colorbar2 
gmt colorbar -Cvelocity.cpt -DJBC+e+mc  -Ba0.4f0.2,+l"S Velocity(km/s)" 

## 绘制 legend 这里使用两种方式 gmt legend & gmt text 分别进行绘制；
gmt legend $R $J -DjTL+w0.4c/0.4c << EOF
H 10p,Times-Roman (a)
EOF
		
echo 124.3 43.0 '(a)' | gmt text $R $J -F+f10p,4,red 

gmt legend $R $J -DjBR+w0.5c/0.5c -F+gwhite << EOF
H 8p,Times-Roman ${depth1}km
EOF

gmt end show

2. 绘制 S 波相对速度切片

通常文章中展现结果时，都是用的相对速度，而非绝对速度，这是因为相对速度对比度更加明显，高低速会更加突出；

下面的例子可以展现了两个不同的反演结果 GMT 绘制速度切片的实例，通过对比可发现，两个反演结果整体趋势一致，但是精细结构存在差异；对于同样的反演结果，在绘图时，生成的网格数据，网格大小的设置，也会一定程度上影响最终的结果展示。

Attention

  • xyz2grd 设置网格 -I ，设置网格越大，最终图像结果会显示更平滑，网格越精细，有可能会看到更小尺度的速度扰动；
  • $R 设置经纬度范围时，需要设置成 -I 最小网格的倍数，否则会 warning，但不影响最终绘图；
  • 下图的两个绘图结果，使用了不同的反演结果，细微处有差别，但是整体趋势一致；

#!/bin/zsh
# Date:2022-02-06 
# Author:Yining Tian 
# relative_anomaly & relative_anomaly_f 是两个不同输入文件，
# 绘图脚本一致，使用不同的数据，绘制出的图像有些许不同，但整体趋势一致；

inp3D="./relative_anomaly" # 使用该目录下的 relative_anomaly 
# inp3D="./relat_anomaly_f"
fig="./Figs/Slice_tyn"
lon1=123
lon2=131.50
# lon2=130.75 
lat1=40 
lat2=45.0
depth=4.0
R="-R"$lon1"/"$lon2"/"$lat1"/"$lat2
J="-JM3i" 
cpt="./cpt"
clip_file="./psclip.xy"
line="../lines/"

gmt begin $fig_$depth png 
    gmt set MAP_FRAME_WIDTH 3p  # width of fancy frame 边框宽度；
    gmt set FONT_ANNOT_PRIMARY 8p,Times-Roman,black # 边框经纬度注释字体；
    gmt set MAP_TICK_LENGTH_PRIMARY 0.025c  

    gmt makecpt -Cblack,red,white,blue,purple -T-14/14/7 -H -Z   > $cpt"/velocity.cpt"
    gmt basemap $R $J -Bya2f1 -Bxa2f1 -BWseN 
    gmt makecpt -Cgray -T-7000/7000/1000 -H -Z > $cpt"/g.cpt"    #cpt grey

    # 绘制地形起伏 
    gmt grdimage @earth_relief_15s $R $J -C$cpt"/g.cpt" -I+d 

    # 在 clip 出的区域，绘制成像结果；
    gmt clip $clip_file $R $J 
    awk '{if($3==depth1) print $1,$2,$5}' depth1="4.0" $inp3D > tmp.dat # relative_anomaly 
    # awk '{if ($4==depth1) print $1,$2,$5}' depth1="4.0" $inp3D > tmp.dat  # relat_anomaly_f
    # gmt xyz2grd $R 的范围应该 0.25 的倍数；否则会 warning 但是不影响正常绘图；
    # -I 生成的网格大，则成像结果会更平滑，网格设置小，则会得到更精细的结果；
    gmt xyz2grd tmp.dat $R $J -I0.25/0.25 -Gtmp.grd 
    # gmt xyz2grd tmp.dat $R $J -I0.5/0.5 -Gtmp.grd 
    gmt grdimage tmp.grd  -C$cpt"/velocity.cpt" $R $J -E720 -Q 
    gmt clip -C 

    # 绘制海岸线
    gmt coast $J $R -N1/0.8p,black,- -W0.8p -A0/0/1 -Dh 

    ## 绘制断裂带 & 火山 使用 -S 选项，则仅在数据点所在位置绘制符号
    gmt plot "./Tanlu.xy" $R $J -W0.5p,black 
    gmt plot "./suture.xy" $R $J -W1p,gray40,- 
    gmt plot "./volcanoes.txt" $R $J -Skvolcano/8p -Gblack  
    
    # 绘制 linea & lineb 
    gmt plot ${line}"linea" $R $J -W1.2p,brown  
    gmt plot ${line}"lineb" $R $J -W1.2p,brown  

    gmt set FONT_LABEL +10p,Times-Roman,black  
    gmt set FONT_ANNOT_PRIMARY 8p,Times-Roman,black # 边框经纬度注释字体；
    gmt colorbar -C$cpt"/velocity.cpt" -DJBC+e+mc  -Ba4f2,+l"Relative Velocity(km/s)" 

    ## 添加 text 统一规定所有文本的字号为 10p(表示字体大小) 4号字体(表示不同字体) -a-10 表示顺时针旋转10° 
    gmt set FONT_ANNOT_PRIMARY 8p,Times-Roman,black
    # gmt text "./volcanoes.txt" $R $J -F+f5p,37,red -D0.05i/-0.12i 
    gmt text "./volcanoes.txt" $R $J -D0.05i/-0.12i 
    echo 126 42.4432 A | gmt text $R $J -F+f12p,4,orange  # 绘制字母 A 
    echo 128.50 42.0017 A\' | gmt text $R $J -F+f12p,4,'#2ca25f'  # 绘制字母 A' 
    echo 127.6 41.2 B | gmt text $R $J -F+f12p,4,orange # 绘制字母 B 
    echo 128.65 44.2 B\' | gmt text $R $J -F+f12p,4,'#2ca25f' # 绘制字母 B' 

# 绘制 legend 
gmt set FONT_ANNOT_PRIMARY 12p,Helvetica FONT_LABEL 12p,Helvetica
gmt legend $R $J -DjRB+w2.0c/0.5c -F+gwhite -t20 -C4p << EOF
H 10p,Times-Roman 4km
EOF

gmt end show
