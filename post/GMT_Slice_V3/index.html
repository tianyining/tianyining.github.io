# Date:2022-03-21 
# Author:Yining Tian 
# Funtion: Using grd2cpt plot Azimuthal anisotropy velocity slice for each depth 

#!/bin/zsh 

input_path="./"
input_file=$input_path"Gc_Gs_model_reSmp.inv"
# input_file=$input_path"Gc_Gs_model.inv" 
output_path="./" 

tomo_cpt=$input_path"BlueDarkRed18.cpt"
fault=$input_path"Fault_Yunnan.txt" 
vsclip=$input_path"YN_clicp.txt"
azmclip=$input_path"Az_clicp.txt"

# GMT Settings 
gmt set MAP_FRAME_TYPE Plain
gmt set MAP_FRAME_PEN 1.2p
gmt set FONT Times-Roman # 同时设置所有 FONT 类参数，
gmt set FONT_ANNOT_PRIMARY 8p,Times-Roman # 设置一级标注的字体；
gmt set FONT_TITLE 15p,Times-Roman # 图上方标题的字体；  
gmt set MAP_TICK_LENGTH_PRIMARY 1.5p/1p # 一级刻度的主刻度和次刻度的长度 [5p/2.5p]
gmt set MAP_TICK_PEN_PRIMARY 1p,black # 一级刻度的画笔属性；
gmt set MAP_TITLE_OFFSET 0p # 图标题的底部与轴标注（或轴标签）的顶部之间的距离 [14p]
gmt set MAP_GRID_PEN_PRIMARY thicker,black # 一级网格的线条属性；
gmt set COLOR_BACKGROUND 0/0/255 # 数据Z值小于CPT文件中最小值时使用的背景色 [black]
gmt set COLOR_FOREGROUND 255/0/0 # 数值 Z 值大于 CPT 文件中最大值时使用的背景色 
gmt set COLOR_NAN 255/255/255 # 数值 Z 为空的时候，所用的颜色值；

fig_name="Az_Depth_Slice_grd2cpt_Subplot"
# fig_name="Az_Depth_Slice_xyz2grd_Subplot_Origin"
fig=${output_path}$fig_name 

gmt begin $fig png  
    gmt subplot begin 2x3 -Fs9.75c/9.5c -B -BWeSn
    R_tomo="-R98/107/21/29"
    J="-Jm1.0c"
    T_range=2.8/4.8 

    gmt subplot set 0 
    gmt basemap $J $R_tomo -Bxa2f2 -Bya2f2 -BWeSn 
    gmt clip $R_tomo $J $vsclip -W0.5p,black,-
    awk '{if($3==depth1) print $1,$2,$4}' depth1='10.0' $input_file > tmp.dat 
    gmt xyz2grd tmp.dat $R_tomo $J -I0.5/0.5 -Gxyz.grd 
    gmt grd2cpt xyz.grd -Cseis -Z -H -D > tmp.cpt 

    gmt grdimage xyz.grd -Ctmp.cpt $R_tomo $J -E720 -Q # 添加 -E720 就会得到和 surface 几乎一致的结果；
    gmt clip -C 
    # plot faults in YunNan Region 
    color=105/105/105 
    gmt plot $fault $R_tomo $J -W1.0p,$color 
    # plot azimuthal anisotropy 
    gmt clip $R_tomo $J $azmclip 
    awk '{if($3==depth1) print $1,$2,$5,$6*40 }' depth1="10.0" $input_file | gmt plot $J $R_tomo -SV0.2c+jc -W0.8p,black -Gred 
    gmt clip -C 
    # plot legend 
    # -1% 
    gmt legend $R_tomo $J -DjRB+w0.8c+o0.1c/0.1c -F+gwhite+p0.4p -t20 << EOF 
H 10p,Times-Roman -1%
EOF
    # (a)
    gmt legend $R_tomo $J -DjLT+w0.5c+o0.05c/0.05c -F+gwhite+p0.35p -t20 << EOF
H 10p,Times-Roman (a)
EOF
    # 5-10km 
    gmt legend $R_tomo $J -DjRT+w1.2c+o0.05c/0.05c -F+gwhite+p0.4p -t20 << EOF
H 10p,Times-Roman 5-10km
EOF
    gmt colorbar -Ctmp.cpt -DJBC+w5c/0.15c+e+mc  -Bxa0.1f0.1 -By+l"Vs (km/s)" 

    gmt subplot set 1 
    gmt basemap $J $R_tomo -Bxa2f2 -Bya2f2 -BWeSn 
    gmt clip $R_tomo $J $vsclip -W0.5p,black,-
    awk '{if($3==depth1) print $1,$2,$4}' depth1='20.0' $input_file > tmp.dat 
    gmt xyz2grd tmp.dat $R_tomo $J -I0.5/0.5 -Gxyz.grd

    gmt clip -C 
    # plot faults in YunNan Region 
    color=105/105/105

    # plot azimuthal anisotropy 
    gmt clip $R_tomo $J $azmclip 
    awk '{if($3==depth1) print $1,$2,$5,$6*40 }' depth1="20.0" $input_file | gmt plot $J $R_tomo -SV0.2c+jc -W0.8p,black -Gred 
    gmt clip -C 
    # plot legend 
    # -1% 
    gmt legend $R_tomo $J -DjRB+w0.8c+o0.1c/0.1c -F+gwhite+p0.4p -t20 << EOF 
H 10p,Times-Roman -1%
EOF
    # (b)
    gmt legend $R_tomo $J -DjLT+w0.5c+o0.05c/0.05c -F+gwhite+p0.35p -t20 << EOF
H 10p,Times-Roman (b)
EOF
    # 15-20km 
    gmt legend $R_tomo $J -DjRT+w1.3c+o0.05c/0.05c -F+gwhite+p0.4p -t20 << EOF
H 10p,Times-Roman 15-20km
EOF
    gmt colorbar -Ctmp.cpt -DJBC+w5c/0.15c+e+mc  -Bxa0.1f0.1 -By+l"Vs (km/s)" 

    gmt subplot set 2
    gmt basemap $J $R_tomo -Bxa2f2 -Bya2f2 -BWeSn 
    gmt clip $R_tomo $J $vsclip -W0.5p,black,-
    awk '{if($3==depth1) print $1,$2,$4}' depth1='30.0' $input_file > tmp.dat 
    gmt xyz2grd tmp.dat $R_tomo $J -I0.5/0.5 -Gxyz.grd

    gmt colorbar -Ctmp.cpt -DJBC+w5c/0.15c+e+mc  -Bxa0.1f0.1 -By+l"Vs (km/s)"
