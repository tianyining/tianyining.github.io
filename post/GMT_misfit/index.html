说明

本篇 blog 用于记录如何使用 GMT 绘制反演结果之间的差异，包含数据格式整理，以及 GMT 绘图代码。

import os
import glob
import numpy as np
import pandas as pd

input_dir = "../DSurfTomo_Results/Named_Depth/"
DSurfTomo_dict = {}
for depth_file in glob.glob(input_dir + "*"):
    depth = os.path.basename(depth_file)

    data = np.loadtxt(input_dir + depth, skiprows=1)
    tmp_d = {}
    for i in range(len(data)):
        line = data[i] 
        lon_depth = line[0]
        lat_depth = line[1]
        depth = line[2]
        vs_depth = line[3]
        key = str(lon_depth) + "_" + str(lat_depth)
        value = vs_depth

        tmp_d[key] = value

        tmp_dict = {float(depth): tmp_d}
    DSurfTomo_dict.update(tmp_dict)

# Right Version

import os
import glob
import numpy as np
import pandas as pd

DSurfTomo_dir = "../DSurfTomo_Results/Named_Depth/"
JointInversion_dir = "../JointInversion_Results/Named_Depth/"
output_dir = "../Inversion_Same_Grid/"

for depth_file in glob.glob(JointInversion_dir + "*"):
    file_name = os.path.basename(depth_file)

    if os.path.exists(DSurfTomo_dir + file_name):
        file_name = float(file_name)

        with open(output_dir + str(file_name), "w") as fileID:
            for key in DSurfTomo_dict[file_name].keys():
                if key in JointInversion_dict[file_name].keys():
                    #                     print(key)  # 共同的 grid

                    lon = key.split("_")[0]
                    lat = key.split("_")[1]

                    Yining_vs = DSurfTomo_dict[file_name][key]
                    Yiming_vs = JointInversion_dict[file_name][key]
                    minus = Yining_vs - Yiming_vs
                    abs_minus = abs(minus) 

                    fileID.write("{:<s}\t{:<s}\t{:<8.4f}\t{:<8.4f}\t{:<8.4f}\t{:<8.4f}\n".format(
                        lon, lat, Yining_vs, Yiming_vs, minus,abs_minus))

#!/bin/zsh 

# Date:2022-05-08 
# Author:Yining Tian 
# Function : 绘制 10km 深度 Yining Tian 和 Yiming Bai 反演结果的差异性； 

output_path="../Figs/"
Sta_Info="../Geo_Infos/STA_Yiming/Sta_Yiming_right"
Volcanoes="../Geo_Infos/volcanoes.lst"

# GMT Settings 
gmt set MAP_FRAME_TYPE Plain 
gmt set MAP_FRAME_PEN 1.0p 
gmt set FONT Times-Roman # 同时设置所有 FONT 类参数，
gmt set FONT_ANNOT_PRIMARY 8p,Times-Roman # 设置一级标注的字体；
gmt set FONT_TITLE 15p,Times-Roman # 图上方标题的字体；  
gmt set MAP_TICK_LENGTH_PRIMARY 1.5p/1p # 一级刻度的主刻度和次刻度的长度 [5p/2.5p]
gmt set MAP_TICK_PEN_PRIMARY 0.5p,black # 一级刻度的画笔属性；
gmt set MAP_TITLE_OFFSET 0p # 图标题的底部与轴标注（或轴标签）的顶部之间的距离 [14p]
gmt set MAP_GRID_PEN_PRIMARY thicker,black # 一级网格的线条属性；
gmt set COLOR_NAN 255/255/255 # 数值 Z 为空的时候，所用的颜色值；

fig_name="Inversion_difference_Subplot1"
fig=$output_path$fig_name 

gmt begin $fig png 
    gmt subplot begin 2x4 -Fs6c -B -BWeSn  
    J_tomo="-JM2i"
    R_tomo="-R92.75/97.5/20.75/24.5" # 比实际 tomo 区域大； 
    gmt subplot set 0 
    input_file="../Inversion_Same_Grid/2.0"
    RMAP=`cat $input_file | gmt info -C | awk '{print "-R"$1"/"$2"/"$3"/"$4}'`
    echo $RMAP 
    gmt basemap $R_tomo $J_tomo -Ba1f0.5 -BWSen 
    cat $input_file | awk '{print $1,$2,$6}' > difference_10.dat 

    gmt xyz2grd difference_10.dat $R_tomo -I0.25/0.25 -Gdifference_10.grd 
    gmt makecpt -Cgray -T0/0.6/0.01 -H -D -I -Z > difference_10.cpt 
    gmt grdimage difference_10.grd -Cdifference_10.cpt $R_tomo $J_tomo 

    gmt coast -EFR,IT+gP300/8 -Di -N1/0.015c,-  $R_tomo $J_tomo 
    cat $Sta_Info | awk '{print $2,$3}' | gmt plot $J_tomo $R_tomo -Si0.035i -W0.2p,black -Gpurple 
    gmt plot $Volcanoes $R_tomo $J_tomo -Skvolcano/5.5p -W0.2p,black -Ggreen 
    gmt set FONT_LABEL +8p,Times-Roman,black  
    gmt set FONT_ANNOT_PRIMARY 6p,Times-Roman,black # 边框经纬度注释字体；
    gmt colorbar -Cdifference_10.cpt  $R_tomo $J_tomo -DJBC+w4c/0.15c+e+mc  -Bxa0.1f0.1+l"Vs Misfit (km/s)"  -By

gmt legend $R_tomo $J_tomo  -DjRT+w1.0c+o0.05c/0.05c -F+gwhite -t20 << EOF
H 8p,Times-Roman 2km 
EOF
