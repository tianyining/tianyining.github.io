说明

本篇 blog 用于记录如何使用 GMT 绘制一些剖面，本篇 blog 仅展示如何绘制海拔剖面、层析成像速度剖面，未完待续。

本次绘制剖面使用数据为 ZGL-2021_EPSL （速度模型未公开）；

感谢 ys 师兄的帮助 ~

import os
import shutil
import numpy as np

file_path = "./data/ZGL_P_velocity"
output_dir_path = "./output_TYN/"

# load the data
data_arr = np.loadtxt(file_path)

if not os.path.exists(output_dir_path):
    os.makedirs(output_dir_path)
else:
    print("output_TYN already exist !!!")

lon_unique_arr = np.unique(data_arr[:, 0])
lon_unique_arr.sort()

lat_unique_arr = np.unique(data_arr[:, 1])
lat_unique_arr

vel_arr = data_arr[:, 3]
depth_unique_arr = np.unique(data_arr[:, 2])
depth_unique_arr.sort()

print("Longitude range:" +
      '{:<} {:<}'.format(min(lon_unique_arr), max(lon_unique_arr)))

print("Latitude range:" +
      '{:<} {:<}'.format(min(lat_unique_arr), max(lat_unique_arr)))
print("Vp range:" + '{:<} {:<}'.format(min(vel_arr), max(vel_arr)))
print("Depth range:" +
      '{:<} {:<}'.format(min(depth_unique_arr), max(depth_unique_arr)))

for i in range(0, len(depth_unique_arr)):  # 共计 11 个 depth
    code = len(depth_unique_arr) - i
    depth = depth_unique_arr[i]
    index_arr = np.where(data_arr[:, 2] == depth)[0]
    target_data_arr = data_arr[index_arr, :]  # 提取特定深度的信息
    new_arr = np.array([target_data_arr[:, 0], target_data_arr[:, 1], 
                       target_data_arr[:, 3], target_data_arr[:, 3]])
    new_arr = new_arr.T
    new_arr = sorted(new_arr, key=lambda x: (x[0], x[1]))

    out_file_path = output_dir_path + "imagep1.lyr{:<}".format(code) 
    np.savetxt(out_file_path, new_arr, fmt="%8.4f %7.4f %8.4f %8.4f")

#!/bin/zsh 

# Date:2022-05-02 
# Author:Yining Tian 
# Functions: Try to plot Vp velocity profile & topo together  

# GMT Settings 
gmt set FONT Times-Roman
gmt set FONT_ANNOT=8p
gmt set FONT_LABEL=12p
gmt set MAP_FRAME_TYPE=plain
gmt set MAP_TICK_LENGTH_PRIMARY=0.02i
gmt set MAP_TICK_PEN_PRIMARY=0.01i
gmt set MAP_FRAME_PEN=0.02i


## Fig a topo profile 
J="-JX7.5i/1.6i"
R_project="-R93/96/22/24" 
RMAP="-R93/96/0/2500"
gmt begin ZGL_Profile_FINAL png 

    # grdtrack topo data 
    gmt basemap $RMAP $J -Bxa1f0.5 -Bya500f1000+l"Elevation (m)" -BWesn
    gmt grdcut @earth_relief_15s $R_project $J -Gtopo_Myanmar.grd 
    gmt project -C93/22.65 -E96/22.65 -G0.01 | gmt grdtrack -Gtopo_Myanmar.grd  > elevationproject_ZGL.tmp
    # 否则 -Ggray 会切一刀； 
    echo "96    0       0       0
    93      0       0       0
    93      23      0       1073" >> elevationproject_ZGL.tmp 

    awk '{print $1,$4}' elevationproject_ZGL.tmp | gmt plot  $RMAP $J -W0.5p,black  -Ggray 

    # 绘制虚线 
    gmt plot -W1p,black,- << EOF
    94.03 0
    94.03 2200
EOF
    gmt text -N -F+f17p,blue <<EOF
    93.25 1750 IBR
    95 650 CB
EOF
#     gmt text -N -F+f15p,cyan <<EOF
#     92.82 2450 a
# EOF


    ## Fig b Vp profile 22.5°N 
    R_big=93/96/21/25
    R=93/96/0/100
    # J=X6c/-1.8c
    J=X7.5i/-1.8i
    cpt="rgb_TYN.cpt"
    cpt_file="./cpt_files/"
    NDEP=11
    DDEP=10 
    layer=1
    DEG1=0.1
    DEG2=0.01

    gmt makecpt -C$cpt_file"seis.cpt" -T3/9/0.01 -H > $cpt 
    gmt project -C93.0/22.65 -E96.0/22.65 -G$DEG2 > gc_ZGL.xyt
    cat gc_ZGL.xyt | awk '{ if ($1 < 0) { print 360.+$1,$2 } else {print $1, $2 }}' | cat > gc_ZGL.xy
    layer=1
	while [ $layer -le $NDEP ]
	do
		depth=`echo $NDEP $layer $DDEP | awk ' { printf("%10.5f\n",($1-$2)*$3+5) } '`
		echo $depth
		gmt surface output_TYN/imagep1.lyr$layer -Gslicep.grd -I0.1/0.1 -R$R_big -T0.1
		gmt grdtrack gc_ZGL.xy -Gslicep.grd > slicep.gc # 需要生成 gc.xy 
		paste gc_ZGL.xyt slicep.gc | awk -vd="$depth" '{print $1, d, $6}' >> slicep.out # 需要生成 gc.xyt 
		layer=`expr $layer + 1`
	done

    gmt surface slicep.out -Gslicep.grd -R$R -I0.01/0.5
    gmt grdimage slicep.grd -Bxa1f0.5+u"\260"+l"Longitude" -Bya20f10+l"Depth (km)" -BnSWe -C$cpt -J$J -R$R -Y-5.0
    gmt colorbar -J$J -R$R_big -C$cpt -DJBC+w8c/0.35c+e+mc  -Bxa1f1+l"Vp (km/s)" -By
    # echo 92.9 0 '(b)' | gmt text -J$J -R$R -F+f10p,4,red 
gmt end show 

rm gmt* *.cpt slicep* gc*
