#!/bin/zsh 
# Date:2022-03-03 
# Author:Yining Tian 
# Version:GMT-6.2.0 

fig="faults_Myanmar_FIANL"
R="-R87.9/108.65/8/31"
J="-JM7c"

# GMT basic settings 
gmt begin $fig png 
    gmt set MAP_FRAME_TYPE fancy # 设置边框类型；
    gmt set MAP_FRAME_WIDTH 1.5p # 设置边框宽窄；
    gmt set FONT_ANNOT_PRIMARY 8p,Times-Roman,black # 边框经纬度注释字体；
    gmt set MAP_TICK_LENGTH_PRIMARY 0.025c # 边框 tick length 经纬度刻度长度；

    gmt coast $R $J -Df -Ggray -Swhite -A10000
    gmt basemap $R $J -Bya4f2 -Bxa4f2 -BWSEN 

    # Plot Sagaing Faults 
    Sagaing_Fault="./faults/Faults-RL-Sagaing.gmt"
    gmt plot $R $J -W1.5p,blue $Sagaing_Fault 

    # Plot RedRiver Faults Main Systems 
    RedRiver_Fault1="./faults/Faults-RL-Red-1.gmt" 
    gmt plot $R $J -W1.5p,blue $RedRiver_Fault1 
    RedRiver_Fault2="./faults/Faults-RL-Red-2.gmt"
    gmt plot $R $J -W1.5p,blue $RedRiver_Fault2 
    RedRiver_Fault3="./faults/Faults-RL-Red-3.gmt"
    gmt plot $R $J -W1.5p,blue $RedRiver_Fault3

    # A separated segment of Red River Faults 
    RedRiver_Fault4="./faults/Faults-RL-Red-4.gmt"
    gmt plot $R $J -W0.3p,blue $RedRiver_Fault4

    # Other right-lateral faults  其他右旋走滑断层 蓝色 
    Faults_RL_Other="./faults/Faults-RL-Other.gmt"
    gmt plot $R $J -W0.3p,blue $Faults_RL_Other  

    # Left-lateral faults purple 左旋走滑断层 紫色
    Faults_LL="./faults/Faults-LL.gmt"
    gmt plot $R $J -W0.3p,purple $Faults_LL 

    # Normal Faults yellow  正断层 黄色
    Faults_Normal="./faults/Faults-N.gmt"
    gmt plot $R $J -W0.3p,yellow $Faults_Normal 

    # Thrust Faults 开始绘制逆冲断层；
    # Himalayan Frontal Thrusts 
    # Faults_Himalayan_1="./faults/Faults-T-Himalayan-1.gmt"
    # gmt plot $R $J -W1.5p,red -Sf0.5c/0.08i+l+t+o0.3c+p0.2p,red -Gred $Faults_Himalayan_1
    # Faults_Himalayan_2="./faults/Faults-T-Himalayan-2.gmt"
    # gmt plot $R $J -W1.5p,red -Sf0.5c/0.08i+l+t+o0.3c+p0.2p,red -Gred $Faults_Himalayan_2

    # Himalayan Frontal Thrusts from Shun Yang 
    Faults_Ocean="./faults/faults_in_ocean_p0.txt"
    gmt plot $R $J -W1.3p,red -Sf0.5c/0.06i+l+t+o0.3c+p0.2p,red -Gred $Faults_Ocean
    # Kabaw Faults 
    Faults_Kabaw="./faults/Faults-T-Kabaw-1.gmt"
    # Faults_Kabaw="./faults/Kabawfault.dat" # Yangshun 
    gmt plot $R $J -W1.0p,red -Sf0.5c/0.05i+r+t+o0.2c+p0.2p,red  -Gred $Faults_Kabaw
    # Arakan megathrust 延续 Himalayan Faults 用绿色标记出；
    Faults_Arakan_1="./faults/Faults-T-arakan-1.gmt"
    # gmt plot $R $J -W1.5p,green -Sf0.6c/0.08i+l+t+o0.5c+p0.2p,green -Ggreen $Faults_Arakan_1
    # Dauki thrust
    Faults_Dauki="./faults/Faults-T-Dauki.gmt"
    # gmt plot $R $J -W1.5p,green -Sf0.5c/0.08i+r+t+o0.2c+p0.2p,green -Ggreen $Faults_Dauki
    # Naga thrust 
    Faults_Naga_1="./faults/Faults-T-Naga-1.gmt"
    Faults_Naga_2="./faults/Faults-T-Naga-2.gmt"
    # gmt plot $R $J -W1.5p,green -Sf0.5c/0.08i+r+t+o0.3c+p0.2p,green -Ggreen $Faults_Naga_1
    # gmt plot $R $J -W1.5p,green -Sf0.5c/0.08i+r+t+o0.3c+p0.2p,green -Ggreen $Faults_Naga_2

    # spreading center 黑色 
    fault_other="./faults/Faults-Other.gmt"
    gmt plot $R $J -W1.0p,black $fault_other 
    
    # Plot Volcano 
    Volcano_dir="../volcanoes.lst"
    gmt plot $Volcano_dir $R $J -Skvolcano/4.5p -W0.1p,black -Gred  

gmt end show
