'''
 Right Version 
 @Author:Yining Tian 
 @Date:2021-11-13 
 @Function:从 Yiming 拿去反演的相速度频散曲线文件中提取台站对的经纬度信息，并写成文件，
 生成 Yiming Sta Info （Yiming 所用的台站经纬度信息）
'''

import os
import glob

input_dir = "../CDisp/"
output_dir = "../Sta_info_Yiming/"
sta_ID = open(output_dir + "sta_info.lst", "w")
for disp_file in glob.glob(input_dir + "*.txt"):
    '''
    读取 disp_file 中的 sta1 sta2 名字
    '''
    disp_file_name = os.path.basename(disp_file)
    sta1 = disp_file_name.split("_")[0].split(
        ".")[2] + "." + disp_file_name.split("_")[0].split(".")[3]
    sta2 = disp_file_name.split("_")[1]
    '''
    读取 disp_file 中对应的 sta1 sta2 经纬度信息
    '''
    fileID = open(disp_file)
    data_info = fileID.readlines()
    fileID.close()
    lon_lat_info = data_info[:2]

    sta2_lon = float(lon_lat_info[0].strip().split()[0])
    sta2_lat = float(lon_lat_info[0].strip().split()[1])
    sta1_lon = float(lon_lat_info[1].strip().split()[0])
    sta1_lat = float(lon_lat_info[1].strip().split()[1])

    sta_ID.write("{:<15s} {:<15.6f} {:<15.6f}\n".format(
        sta1, sta1_lon, sta1_lat))
    sta_ID.write("{:<15s} {:<15.6f} {:<15.6f}\n".format(
        sta2, sta2_lon, sta2_lat))
    sta_ID.flush()

import os
import glob

input_dir = "../CDisp/"
output_dir = "../CDisp_Only/CDisp_raw/"

for disp_file in glob.glob(input_dir + "*.txt"):
    #     print(disp_file)
    disp_file_name = os.path.basename(disp_file)
    new_disp_file_name = disp_file_name[:-11] + ".txt"

    fileID = open(disp_file)
    data_info = fileID.readlines()
    fileID.close()
    data_info_new = data_info[2:]

    new_disp = open(output_dir + new_disp_file_name, "w")
    for item in data_info_new:
        new_disp.write(item)
        new_disp.flush()

'''
该版本使用的是 python list 的读法，没有使用 numpy pandas 等包，运行速度较慢 1m30s 左右
'''
import os
import glob
import numpy as np
import matplotlib.pyplot as plt
from jupyterthemes import jtplot

jtplot.style(theme="monokai")

input_dir = "../CDisp/"
output_dir = "../Figs/"

count = 0
CCDispFiles = glob.glob(input_dir + "*.txt")
plt.figure(figsize=(12, 6.5))
for disp_file in CCDispFiles:
    count += 1
    sgl_disp = open(disp_file)
    sgl_disp_data = sgl_disp.readlines()
    sgl_period = []
    sgl_phv = []

    for line in sgl_disp_data[2:]:
        if line.split()[1] == '0.000':
            continue
        sgl_period.append(float(line.split()[0]))
        sgl_phv.append(float(line.split()[1]))
        # 设置坐标轴范围
        #         plt.xlim((1.5, 40))
        #         plt.ylim((1.5, 5))

        plt.title('PhV Dispersion Curves Del SNR',
                  fontsize='large',
                  fontweight='normal')

        plt.plot(sgl_period, sgl_phv, 'r-')
plt.savefig(
    "/home/tianyining/Practice/MM_Whole/Yiming/Figs/Yiming_BeforeQC.png")
