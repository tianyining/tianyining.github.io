@Author:Yining Tian

@Function:实现批量处理，将所有 CSV 中的两列数据（period phV）读取出来，并按行写入到 txt 文件；
'''

import os
import glob
import pandas as pd

input_dir = "../input/step2_del_snr/"
output_dir = "../output/MergeFile_MM/"

test_file = open(output_dir + "MergeFile_MM_break.txt", "w")
csvfiles = glob.glob(input_dir + "*.csv")
fmt1 = "{:<10s} {:<10s}\n"
fmt2 = ">\n"
for csvfile in csvfiles:
    #     print(csvfile)
    sgl_disp = open(csvfile, "r")
    sgl_disp_length = len(sgl_disp.readlines())

    df = pd.read_csv(csvfile)
    data_period = pd.read_csv(csvfile, sep=",")["period"].values
    data_phV = pd.read_csv(csvfile, sep=",")["ph_V"].values

    for i in range(0, sgl_disp_length - 1):
        test_file.write(fmt1.format(str(data_period[i]), str(data_phV[i])))
    test_file.write(fmt2.format())
test_file.close()
