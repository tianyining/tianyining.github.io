说明

利用 AFTAN 提取出频散曲线后，经过前两步质量控制后：1) 剔除挑错支的频散； 2) 剔除掉低信噪比的相速度数据点。对于频散数据的离群点需要进一步剔除，本篇 blog 使用标准差的筛选方式（通过控制偏离平均值筛选数据点）对相速度频散数据点进行筛选；

之前的 blog 使用箱型图的方法筛选相速度频散数据 ，本质上是用中位数代表总体评价水平，而本篇 blog 的标准差筛选准则是基于数据相对于平均值的偏离程度。

STEP1 MergeFile

将 QCSTEP2 中去除低信噪比数据点后的所有频散文件，整理成一个 disper_curves_compute 文件，文件包含四列：period phaseV GroupV SNR 。

这样做的目的是为后面计算不同周期的相（群）速度标准差做准备。

'''
 Version : 2 * std 
 计算不同周期的相速度平均值、对应的标准差、以及统计每个周期对应有多少个相速度值
'''
import os
import glob
import math
import numpy as np

input_dir = "../MergeFile/"
output_dir = "../Myanmar_Statistcs_File/"

data = np.loadtxt(input_dir + "Disper_Curves_Merge")
shape = data.shape
'''
 eatablish dict 
'''
dict_disp = {}
d = {}  # the value of d is standard deviation 不同周期对应标准差
a = {}  # the value of a is average velocity for each period 不同周期相速度平均值
s = {}  # the value of s is sum phaseV of each period
n = {}  # data point of each period
ss = {}  # sum of square(real phaseV - average)
PhaseV_min = {}
PhaseV_max = {}
'''
 index i 
'''
for i in range(shape[0]):
    period = int(data[i][0])
    if period not in s.keys():
        s[period] = data[i][2]
        n[period] = 1
    else:
        s[period] = s[period] + data[i][2]
        n[period] += 1
    if i == shape[0] - 1:
        for item in s.keys():
            a[item] = s[item] / n[item]
# 对 a s n 按照 key 的大小进行排序(key=period)
a = dict(sorted(a.items(), key=lambda x: x[0]))
s = dict(sorted(s.items(), key=lambda x: x[0]))
n = dict(sorted(n.items(), key=lambda x: x[0]))
'''
 index j 
'''
for j in range(shape[0]):
    period = int(data[j][0])
    if period not in ss.keys():
        ss[period] = math.pow((data[j][2] - a[period]), 2)
    else:
        ss[period] = math.pow((data[j][2] - a[period]), 2) + ss[period]
    if j == shape[0] - 1:
        for item in ss.keys():
            d[item] = math.sqrt(ss[item] / n[item])
ss = dict(sorted(ss.items(), key=lambda x: x[0]))
d = dict(sorted(d.items(), key=lambda x: x[0]))
'''
 index k 偏离 2 倍标准差；
 2*sigma
'''
for k in range(shape[0]):
    period = int(data[k][0])
    if period not in PhaseV_min.keys():
        PhaseV_min[period] = a[period] - 2 * d[period]
        PhaseV_max[period] = a[period] + 2 * d[period]
PhaseV_min = dict(sorted(PhaseV_min.items(), key=lambda x: x[0]))
PhaseV_max = dict(sorted(PhaseV_max.items(), key=lambda x: x[0]))
'''
 将计算结果写入新的统计文件
 All_Period_PhaseV_Scope
'''
with open(output_dir + "All_Period_PhaseV_Scope_2std", "w") as fp:
    for p in PhaseV_min.keys():
        fp.write("{:<d}\t{:<4f}\t{:<4f}\t{:<4f}\t{:<4f}\n".format(
            p, a[p], d[p], PhaseV_min[p], PhaseV_max[p]))

'''
 Version : 3 * std 
 计算不同周期的相速度平均值、对应的标准差、以及统计每个周期对应有多少个相速度值
'''
import os
import glob
import math
import numpy as np

input_dir = "../MergeFile/"
output_dir = "../Myanmar_Statistcs_File/"

data = np.loadtxt(input_dir + "Disper_Curves_Merge")
shape = data.shape
'''
 eatablish dict 
'''
dict_disp = {}
d = {}  # the value of d is standard deviation 不同周期对应标准差
a = {}  # the value of a is average velocity for each period 不同周期相速度平均值
s = {}  # the value of s is sum phaseV of each period
n = {}  # data point of each period
ss = {}  # sum of square(real phaseV - average)
PhaseV_min = {}
PhaseV_max = {}
'''
 index i 
'''
for i in range(shape[0]):
    period = int(data[i][0])
    if period not in s.keys():
        s[period] = data[i][2]
        n[period] = 1
    else:
        s[period] = s[period] + data[i][2]
        n[period] += 1
    if i == shape[0] - 1:
        for item in s.keys():
            a[item] = s[item] / n[item]
# 对 a s n 按照 key 的大小进行排序(key=period)
a = dict(sorted(a.items(), key=lambda x: x[0]))
s = dict(sorted(s.items(), key=lambda x: x[0]))
n = dict(sorted(n.items(), key=lambda x: x[0]))
'''
 index j 
'''
for j in range(shape[0]):
    period = int(data[j][0])
    if period not in ss.keys():
        ss[period] = math.pow((data[j][2] - a[period]), 2)
    else:
        ss[period] = math.pow((data[j][2] - a[period]), 2) + ss[period]
    if j == shape[0] - 1:
        for item in ss.keys():
            d[item] = math.sqrt(ss[item] / n[item])
ss = dict(sorted(ss.items(), key=lambda x: x[0]))
d = dict(sorted(d.items(), key=lambda x: x[0]))
'''
 index k 偏离 2 倍标准差；
 2*sigma
'''
for k in range(shape[0]):
    period = int(data[k][0])
    if period not in PhaseV_min.keys():
        PhaseV_min[period] = a[period] - 3 * d[period]
        PhaseV_max[period] = a[period] + 3 * d[period]
PhaseV_min = dict(sorted(PhaseV_min.items(), key=lambda x: x[0]))
PhaseV_max = dict(sorted(PhaseV_max.items(), key=lambda x: x[0]))
'''
 将计算结果写入新的统计文件
 All_Period_PhaseV_Scope
'''
with open(output_dir + "All_Period_PhaseV_Scope_3std", "w") as fp:
    for p in PhaseV_min.keys():
        fp.write("{:<d}\t{:<4f}\t{:<4f}\t{:<4f}\t{:<4f}\n".format(
            p, a[p], d[p], PhaseV_min[p], PhaseV_max[p]))

'''
 按照 2 倍标准差文件 All_Period_PhaseV_Scope_2std 建立字典
 规定不同周期 PhaseV 的阈值范围
'''
import os
import glob
import numpy as np
import pandas as pd

input_disp_dir = "../Del_SNR_DISP/"
output_disp_dir = "../Del_STD_Phase_2std/"
input_statistics_file = "../Myanmar_Statistcs_File/"
'''
 mkdir 
'''
if not os.path.exists(output_disp_dir):
    os.mkdir(output_disp_dir)
else:
    print("output_disp_dir all ready exists !!!")
'''
 eatablish dict firstly 
'''
namescol = ["period", "average_phaseV", "std", "PhaseV_min", "PhaseV_max"]
dataset = pd.read_table(input_statistics_file + "All_Period_PhaseV_Scope_2std",
                        delim_whitespace="\t",
                        names=namescol)
PhaseV_min = list(dataset["PhaseV_min"])
PhaseV_max = list(dataset["PhaseV_max"])
Period = list(dataset["period"])

PhaseVmin_dict = dict(zip(Period, PhaseV_min))
PhaseVmax_dict = dict(zip(Period, PhaseV_max))
'''
 使用 pandas 对每个台站对的频散文件进行数值筛选： 2 * std 
'''
period_select = np.arange(2, 104, 1)
for sgl_disp in glob.glob(input_disp_dir + "*_Disp_QC2.txt"):
    sgl_disp_name = os.path.basename(sgl_disp)
    namescol = ["Period", "GroupV", "PhaseV", "SNR"]
    df = pd.read_csv(
        sgl_disp,
        sep="\s+",  # 使用正则表达式；
        names=namescol,
        header=None,
    )
    '''
    筛选 & 将筛选后的台站对频散文件重新保存为新的频散文件；
    '''
    str = '_'
    new_disp_name = str.join(
        sgl_disp_name.strip().split(".")[0].split("_")[:-3])
    with open(output_disp_dir + new_disp_name + "_Del_2std", "w") as fp:
        for item in period_select:
            if item in df["Period"].values:
                #                 print(item)
                '''
                 读取 select_period 周期的 PhaseV 数值
                '''
                PhaseV_part = df[df["Period"] == item]['PhaseV'].values[0]
                '''
                 判断该数值 PhaseV_part 是否在对应周期的 2 * std 范围内？
                '''
                if PhaseVmin_dict[item] <= PhaseV_part <= PhaseVmax_dict[item]:
                    fp.write("{:<d}\t{:<4f}\n".format(item, PhaseV_part))
                    fp.flush()
                else:
                    pass

import os
import glob
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from jupyterthemes import jtplot

jtplot.style(theme="monokai")

count = 0
plt.rcParams["axes.linewidth"] = 1  # 图框宽度；
plt.rcParams["figure.dpi"] = 720  #
plt.rcParams['font.sans-serif'] = ['Times New Roman']
plt.rcParams['legend.title_fontsize'] = '20'

output_dir = "../Figs/"

 设置图片
'''
plt.figure(figsize=(12, 8))
plt.tick_params(labelsize=20)
plt.xlabel("Period(s)", size=20)
plt.ylabel("PhaseV(km/s)", size=20)
plt.grid(
    linestyle="--",
    alpha=0.3,
    color="grey",
)

count = 0
disp_files = glob.glob(input_dir + "*_2std")
for sgl_disp in disp_files[:-1]:
    sgl_disp_name = os.path.basename(sgl_disp)
    count += 1
    namescol = ["Period", "PhaseV"]
    df = pd.read_csv(
        sgl_disp,
        sep="\s+",
        names=namescol,
        header=None,
    )
    Period = df.Period.values
    PhaseV = df.PhaseV.values
    '''
     Plot Begin !!! 
    '''
    plt.title("All Period PhaseV Scatter Plot After Del STD", size=26)
    plt.scatter(
        Period,
        PhaseV,
        color="darkviolet",
        s=4,
        edgecolors="black",
    )

final = pd.read_csv(disp_files[-1], sep="\s+", names=namescol, header=None)

Period = final.Period.values
PhaseV = final.PhaseV.values
plt.scatter(Period, PhaseV, s=30, c="brown", label="2*STD")

plt.legend(
    title=("Standard Deviation"),
    loc="best",
    edgecolor="grey",
    fontsize=20,
    fancybox=True,
    shadow=False,
)

 Savefig 
'''
