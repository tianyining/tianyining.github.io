首先需要将这三类台站从总的台站信息文件 station_mm.lst 中分离统计出，并分别存为文件；

'''
 根据台站来源，对 116 个台站进行划分
 1. CMGSMO 缅甸一期台站 46 个；
 2. IRIS 19 个；
 3. YN 云南固定台站 51 个；
'''

import os 
import glob
import pandas as pd
'''
 Read station_mm.lst Info  
'''
input_sta_info = "../STA/station_mm.lst"
output_sta_dir = "../STA/"

namescol = ["Sta_name", "Sta_lon", "Sta_lat"]
Sta_data = pd.read_table(input_sta_info, sep="\s+", names=namescol)
data1 = Sta_data[0:2]
data1.to_csv(
    output_sta_dir + "None/None_Stainfo",
    index=False,
    sep="\t",
    header=None,
    float_format='%.6f',
)

CMGSMO = Sta_data[2:48]
CMGSMO.to_csv(
    output_sta_dir + "CMGSMO/CMGSMO_Stainfo",
    index=False,
    sep="\t",
    header=None,
    float_format="%.6f",
)

IRIS = Sta_data[48:65]
IRIS.to_csv(
    output_sta_dir + "IRIS/IRIS_Stainfo",
    index=False,
    sep="\t",
    header=None,
    float_format="%.6f",
)

YN = Sta_data[65::]
YN.to_csv(output_sta_dir + "YN/YN_Stainfo",
          index=False,
          sep="\t",
          header=None,
          float_format="%.6f",)

需要注意的是，这里划分出 data1 的两个台站，实际上属于 IRIS ，手动将 SIMF、SIMY 两个台站信息粘贴到 IRIS 台站信息中即可；

#!/bin/zsh 

import pygmt
import os
import glob
import pandas as pd
from jupyterthemes import jtplot

jtplot.style(theme="monokai")

input_sta_dir = "../STA/"
'''
 Plot BEGIN !!! Basemap 
'''

G_IRIS = "255/215/0" # 黄色
G_CMGSMO = "65/105/225" # 蓝色
G_YN1 = "255/127/80" # 橙色
G_YN2 = "253/139/254" # 粉色；
fig = pygmt.Figure()

fig.grdimage(
    "@earth_relief_15s",
    region=[89, 107, 16, 29],
    projection='m1:35000000',  # 
    #     projection= "M3i",
    cmap="../Figure_Data/CPT/topo.cpt")

pygmt.config(
    MAP_FRAME_TYPE='plain',  # 不添加黑白相间的 frame 边框；
    MAP_FRAME_WIDTH='1.5p',  # 绘制边框的画笔粗细 2p 
    FORMAT_GEO_MAP='D',  # 设置经纬度表示方式，不显示 E N，只显示浮点数；
    FONT_ANNOT_PRIMARY="6p,Helvetica",
    
)
fig.coast(
    resolution='f',
    shorelines='1/thinnest,black',
    frame=['x5', 'y5', 'NeSW'],
    borders='1/0.25p,-',  # 用虚线表示国界线；
)


'''
Plot Stations IRIS 
'''
'''
 IRIS 
'''
namescol = ["Sta_Name", "Sta_lon", "Sta_lat"]
data1 = pd.read_csv(
    input_sta_dir + "IRIS/IRIS_Stainfo",
    sep="\s+",
    names=namescol,
)
fig.plot(
    x=data1.Sta_lon,
    y=data1.Sta_lat,
    style="t0.09c",
    pen="0.008c,black",
    color=G_IRIS,
    label="IRIS",
)

'''
 CMGSMO 
'''
namescol = ["Sta_Name", "Sta_lon", "Sta_lat"]
CMGSMO = pd.read_csv(
    input_sta_dir + "CMGSMO/CMGSMO_Stainfo",
    sep="\s+",
    names=namescol,
)
fig.plot(
    x=CMGSMO.Sta_lon,
    y=CMGSMO.Sta_lat,
    style="t0.09c",
    pen="0.008c,black",
    # color="dodgerblue",
    color=G_CMGSMO,
    label="CMGSMO",
)

'''
YN
'''
namescol = ["Sta_Name", "Sta_lon", "Sta_lat"]
YN = pd.read_csv(
    input_sta_dir + "YN/YN_Stainfo",
    sep="\s+",
    names=namescol,
)
fig.plot(
    x=YN.Sta_lon,
    y=YN.Sta_lat,
    style="t0.09c",
    pen="0.008c,black",
    color="red2",
    label="YN",
)

'''
 Add legend
'''
with pygmt.config(FONT_ANNOT_PRIMARY="5.5p,Helvetica",
                  FONT_LABEL="5.5",
                  MAP_FRAME_PEN="0.6p"):
    fig.legend(
        position="JBR+jBR+o0.1c",
        # box="+plinen+glinen+gwhite@20+p0.5p,black",
        box='+gwhite+p0.5p+gwhite@20',
    )


'''
 添加 colorbar 表示 topography 
'''

with pygmt.config(
        FONT_ANNOT_PRIMARY='5.5p,Helvetica',
        FONT_LABEL='5.5p',
        MAP_ANNOT_OFFSET_PRIMARY='0.05i',
        MAP_TICK_LENGTH_PRIMARY='0.05',
        MAP_LABEL_OFFSET='3.5p',
        MAP_FRAME_PEN='0.8p',
):
    fig.colorbar(cmap = 'topo.cpt',
                #  position="x0.1i/0.7i+w1.2i/0.075i", # w高度/长度；  彩条 的
                #  position="x0.1i/0.7i+w0.2i/0.075i",
                #  position="x0.1i/0.7i+w0.8i/0.065i",
                 position="JMR+w1.4i/0.075i",
                 frame=['x1500+l"Topography(m)"'],
                 box = '+pwhite+gwhite@20+p0.5p',
                 
                 )
'''
 添加断层信息
'''
style_RD = "f6c/0.1i+r+s30+o2.25c -W1.0p,black"  # Right

fig.plot(
    data="../Figure_Data/RedRiver/RRF.txt",
    pen="0.5p,black",
    style=style_RD,
    #     font="Courier-Bold",
)

style_SF = "f6c/0.1i+r+s30+o2.25c -W1.0p,black"  # Right
fig.plot(
    data="../Figure_Data/SF/SF.txt",
    pen="0.5p,black",
    style=style_SF,
)
fig.plot(data="../Figure_Data/SF/SF_10.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_2.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_3.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_4.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_5.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_6.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_7.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_8.txt", pen="0.5,black")
fig.plot(data="../Figure_Data/SF/SF_9.txt", pen="0.5,black")

'''
 Add Faults Text 
'''
fig.text(
    textfiles=None,
    x=96.5,
    y=21.9,
    position=None,
    text='Sagaing Fault',
    angle=-81.8,
    font='5.5p,Helvetica-Bold,black',
    justify='LM',
)

fig.text(
    textfiles=None,
    x=103.0,
    y=22.3,
    position=None,
    text='RedRiver Fault',
    angle=-42,
    font='5.5p,Helvetica-Bold,black',
    justify='BL',
)

'''
 Plot faults_in_ocean.txt
'''
# style = "f1c/0.1i+l+t -Gblack -W -B10" # pygmt
style = "f0.7+t+l+o0.1 -Gblack" # yangshun 
fig.plot(
    data="../Figure_Data/faults_in_ocean_p0.txt",
    style=style,
    pen="0.6,black",
)

'''
 Visulization 
'''
fig.show()
fig.savefig("../Figs/FINAL/Sta_Region.png",dpi=720)
fig.savefig("../Figs/FINAL/Sta_Region.pdf",dpi=720)

#!/bin/zsh

# Author:Yining Tian 
# Date:2022-02-02
# Function:plot eq & MM_2 Station distribution 
# GMT 6.2.0 

fig="./Figs/plot_eq_sta_MM2_FINAL"
# Data 
topo="extra_gd/Wangxi_Figure1/Block/topo.cpt"
DATA="extra_gd/Wangxi_Figure1"
extra_gd_path="./extra_gd"
sagaingfault="extra_gd/Sagaingfault.dat"
kabawfault="extra_gd/Kabawfault.dat"
eqfile="./eq/catalog_sel_1901-2008_after_relo000.txt" 
station_tyn="/home/edentian/Myanmar_ANT/plot_eq_tyn/Station_Info/station.mm_right"
station_jmm="./Station_Info/stn_M2"

# parameters related to the horizontal slices.
JJ=M9.0c
RR=91.5/101/16/24.5 

# GMT plot Begin !!! 
gmt begin $fig png,pdf
    gmt set MAP_FRAME_WIDTH 2p  # width of fancy frame 边框宽度；
    gmt set MAP_FRAME_PEN 0.5p # pen width of plain frame 
    gmt set FONT_ANNOT_PRIMARY 10p,Times-Roman,black # primary annotation font 边框注释字体大小;
    gmt set MAP_TICK_LENGTH_PRIMARY 0.025c # Length of primary tick marks colorbar & 边框刻度大小
    gmt set FONT_LABEL +8p,Times-Roman,red # colorbar 注释字体

    # 绘制地形图 
    gmt basemap -R$RR -J$JJ -Bxa2f1WSen -Bya2f1WSen 
    gmt grdimage @earth_relief_15s -R$RR -J$JJ -C$topo -I 

    # -t 设置透明度，range 0~100 默认是0,为不透明；
    cat $DATA/Block/IndoBurma.txt | gmt plot -J$JJ -R$RR -G219/214/169 -t60
    cat $DATA/Block/CMB.txt | gmt plot -J$JJ -R$RR -G162/200/185 -t60
    cat $DATA/Block/ShanTaiBlock.txt | gmt plot -J$JJ -R$RR -G204/204/204 -t80

    cat $DATA/Block/ST2.txt | gmt plot -J$JJ -R$RR -G204/204/204 -t80
    cat $DATA/Block/ST3.txt | gmt plot -J$JJ -R$RR -G204/204/204 -t80 
    cat $DATA/Block/MMB.txt | gmt plot -J$JJ -R$RR -G204/153/102 -t60 

    # 绘制海岸线；
    gmt coast -J$JJ -R$RR -W1p,black -Di -S99/102/106

    # 绘制构造线 
    # cat $extra_gd_path/Major_faults.gmt | gmt plot -J$JJ -R$RR -W1.0p 
    cat $extra_gd_path/Sagaingfault.dat | gmt plot -J$JJ -R$RR -Sf4/0.6+r+s+o0.9i -W1.0p,black 
    # cat $extra_gd_path/Shanscarpfault.dat | gmt plot -J$JJ -R$RR -W1.0p,blue 
    # cat $extra_gd_path/Myanmar_newfaults_CL.gmt | gmt plot -J$JJ -R$RR -W1.5p,blue
    # cat $extra_gd_path/KF_a | gmt plot -J$JJ -R$RR -W1.5p,blue 
    gmt plot -J$JJ -R$RR -W1,red -Gred -Sf0.5/0.05i+r+t $kabawfault

    # plot eq  
    cat $eqfile | awk '{print $1,$2,$3,0.038}' | gmt plot -J$JJ -R$RR -Sci -Ccolor.cpt -W0.6p,black 
    
# gmt text 
gmt text -J$JJ -R$RR -F+f+a+j -Gwhite -: >> $fig << EOF
16.57 98.90 7 0 1 Focal Depth
16.2 96.16 8 92 1 Sagaing Fault
23.0 93.80 8 77 1 Kabaw Fault
16.57 97.80 8 0 1 City
EOF

# plot station 
# awk '{print $2,$3}' $station_tyn | gmt plot -Si0.075i -W0.4p,black -Gblue  
cat $station_jmm | awk '{print $4,$3}' | gmt plot -J$JJ -R$RR -Si0.075i -W0.4p,black -Gpurple

# plot city 
echo 95.43 22.77 0.28 | gmt plot -J$JJ -R$RR -Sh0.10i -W0.5p,black -Gyellow 
echo 94.85 22.43 0.28 | gmt plot -J$JJ -R$RR -Sh0.10i -W0.5p,black -Gyellow 

# Legend Begin !!! 绘制 legend 使用两种形式 gmt legend & gmt text 
# plot colorbar 
# gmt colorbar -Dx3.0i/0.18i/0.5i/0.07ih -Ccolor.cpt -I -Bxa50f25+l"km" 
gmt set FONT_ANNOT_PRIMARY 4p,Times-Roman,black # 设置 colorbar 的字体；
gmt colorbar -Dx3.0i/0.18i/0.5i/0.07ih -Ccolor.cpt -I -Bxa50f25
# city legend 
echo 98.0 16.39 0.3 | gmt plot -J$JJ -R$RR -Sh0.10i -W0.5p,black -Gyellow 

# legend for others 
C1=204/153/102
C4=219/214/169
C5=162/200/185
C6=204/204/204
gmt set FONT_ANNOT_PRIMARY 6p,Times-Roman,black 
gmt legend -J$JJ -R$RR -DjBL+w1.25i/0.7i+l1.5+o0.01i/0.01i -C0.1i/0.05i -F+gazure4 >> $fig << EOF
S 0.05i -        0.18i black               1.0p,black                0.27i  Fault zones
S 0.05i r        0.12i $C4@60              1.0p,$C4@60               0.27i  Indo-Burma Ranges
S 0.05i r        0.12i $C5@60              1.0p,$C5@60               0.27i  Central Lowland
S 0.05i r        0.12i $C1@60              1.0p,$C1@60               0.27i  Mogok Metamorphic belt
S 0.05i r        0.12i $C6@60              1.0p,$C6@60               0.27i  Shan Plateau
EOF

gmt end show
