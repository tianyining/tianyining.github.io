Making a 2-D map of surface wave velocities

There are two further steps to go from a set of dispersion curves to a 3-D map of the subsurface. The first step is to make a 2-D period map. When we measure a dispersion curve between two stations. We are observing the effects of the subsurface over the entire distance along these two stations. The purpose of the 2-D period map is to determine where along this path the variations in velocity occur. Here I’ll use seismic-noise-tomography, produced by Dr. Bruno Gotourbes as the framework for developing these 2-D period maps. I’ll go through the tomography workflow step by step, although I have also included a wrapper function called invert4model in rosesutils that can be used to do the same thing but more concisely.

import glob
import os
import obspy
import pandas as pd

input_dir = "./COR-MM/"
output_dir = "./outputfiles/"

fmt = "{:<10s} {:<10s} {:<10.4f}\n"

COR_MM = glob.glob(input_dir + "*.SAC")
with open(output_dir + "COR_MM_dist", "w") as fileID:
    for sgl_cor in glob.glob(input_dir + "*.SAC"):
        sgl_cor_name = os.path.basename(sgl_cor)
        sta1 = sgl_cor_name.split(".")[0].split("_")[1]
        sta2 = sgl_cor_name.split(".")[0].split("_")[2]
        trace = obspy.read(sgl_cor)

        sgl_cor_dist = trace[0].stats.sac.dist

        fileID.write(fmt.format(sta1, sta2, sgl_cor_dist))
fileID.close()

利用 pyproj 库，计算两个台站间的 distance ，其中 pyproj 库可以进行坐标系转换、方位角计算、两点之间距离计算等。

import pyproj

wgs84 = pyproj.Geod(ellps='WGS84')
def dist(lons1, lats1, lons2, lats2):
    """
    Returns an array of geodetic distance(s) in km between
    points (lon1, lat1) and (lon2, lat2)
    """
    _, _, d = wgs84.inv(lons1=lons1, lats1=lats1, lons2=lons2, lats2=lats2)
    return np.array(d) / 1000.0

import os
import glob
import pandas as pd

input_dir = "./COR-MM/"
output_dir = "./outputfiles/"
fmt = "{:<10s} {:<10s} {:<10.4f} {:<10.4f} {:<10.4f} {:<10.4f}\n"

        sta2 = sgl_cor_name.split(".")[0].split("_")[2]

        sta1_lon = Sta_Info[sta1]["sta_lon"]
        sta1_lat = Sta_Info[sta1]["sta_lat"]
        sta2_lon = Sta_Info[sta2]["sta_lon"]
        sta2_lat = Sta_Info[sta2]["sta_lat"]

        fileID.write(
            fmt.format(sta1, sta2, sta1_lon, sta1_lat, sta2_lon, sta2_lat))
fileID.close()

import os
import glob
import pandas as pd

input_file = "./outputfiles/Stainfo_MM"
output_dir = "./outputfiles/"
namescol = ["Sta1", "Sta2", "Sta1_lon", "Sta1_lat", "Sta2_lon", "Sta2_lat"]

data = pd.read_csv(input_file, names=namescol, sep="\s+")

with open(output_dir + "Cal_dist_pyproj_MM","w") as fileID:

    for index, row in data.iterrows():
        dist_MM = dist(row.Sta1_lon, row.Sta1_lat, row.Sta2_lon, row.Sta2_lat)
        sta1 = row.Sta1
        sta2 = row.Sta2
        fileID.write("{:<10s} {:<10s} {:<10.4f}\n".format(sta1, sta2, dist_MM))
