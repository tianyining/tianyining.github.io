import os
import pandas as pd
import numpy as np

input_dir = "../Period_Disp/"

namescol = [
    "source_name", "receiver_name", "group_velocity", "phase_velocity",
    "signal2noise_ratio", "source_lon", "source_lat", "receiver_lon",
    "receiver_lat", "inter-station_distance"
]

disp_curves = pd.read_csv( 
    input_dir + "10.dat",  
    names=namescol,
    delim_whitespace=True, 
)

# Clean Data Delete Outliers
v_type = "phase_velocity"
origin = len(disp_curves)
std_crit = np.abs(disp_curves[v_type] - np.mean(disp_curves[v_type])
                  ) < 3 * np.std(disp_curves[v_type])

disp_curves_clean = disp_curves[std_crit]
clean = len(disp_curves_clean)

print(f"{origin - clean} measurements were removed")

import pyproj 

wgs84 = pyproj.Geod(ellps="WGS84")

def dist(lons1,lats1,lons2,lats2):
    """
    Returns an array of geodetic distance(s) in km between
    points (lon1, lat1) and (lon2, lat2)
    """
    _, _, d = wgs84.inv(lons1=lons1, lats1=lats1, lons2=lons2, lats2=lats2)
    return np.array(d) / 1000.0

def geodesic(coord1, coord2, npts):
    """
    Returns a list of *npts* points along the geodesic between
    (and including) *coord1* and *coord2*, in an array of
    shape (*npts*, 2).
    @rtype: L{ndarray}
    """
    if npts < 2:
        raise Exception('nb of points must be at least 2')

    path = wgs84.npts(lon1=coord1[0], lat1=coord1[1],
                      lon2=coord2[0], lat2=coord2[1],
                      npts=npts-2)
    return np.array([coord1] + path + [coord2])

'''
Modified from seismic-noise-tomography 
Calculate Ray Paths 
'''
paths = []  # 20s 周期，测量不同台站对间的频散，有多少个台站对，就有多少条 path
dists = []  # 20s 周期，测量不同台站对间的 dist_test 有多少个台站对，就有多少个 dist_test
coords = []

path_num = 0
for index, row in disp_curves_clean.iterrows():

    # calculate the distance between two points(two stations)
    dist_test = dist(row.source_lon, row.source_lat, row.receiver_lon,
                     row.receiver_lat)
    '''
    npts 当 dist_test > 100 时，为 dist_test 本身，否则为 100 
    '''
    npts = np.max([np.ceil(dist_test) + 1, 100])

    source_coords = (row.source_lon, row.source_lat)
    if source_coords not in coords:
        coords.append(source_coords)
    receiver_coords = (row.receiver_lon, row.receiver_lat)
    if receiver_coords not in coords:
        coords.append(receiver_coords)
    '''
    Calculate the path between source_coords & receiver_coords 
    '''
    path = geodesic(source_coords, receiver_coords, npts)
    path_num += 1
    paths.append(path)
    dists.append(dist_test)
'''
 number of dists : 323 
 path_num = 323 
 
'''
dists = np.array(dists)
paths = np.array(paths, dtype="object")

len(dists) == len(paths)

'''
 使用面向对象的编程绘图方式； 以 10s 周期为例，绘制提取出频散数据的台站对射线分布图； 
'''
import matplotlib.pyplot as plt
import matplotlib.font_manager as font_manager
from matplotlib.lines import Line2D

fig, ax = plt.subplots(
    figsize=(6, 6),
    dpi=720,
)

for path in paths:
    plt.plot(path.T[0], path.T[1], color="black", alpha=0.2)
for coord in coords:
    plt.plot(
        coord[0],
        coord[1],
        marker="v",
        color="yellow",
        markersize=8,
        markeredgecolor="black",
    )
'''
 Create custom legend
'''
font = font_manager.FontProperties(family='Times New Roman',
                                   weight='normal',
                                   style='normal',
                                   size=10)
my_legend = [
    Line2D([0], [0], color="black", label="Raypath", alpha=0.2),
    Line2D(
        [0],
        [0],
        color="yellow",
        marker="v",
        markeredgecolor="black",
        markersize=8,
        lw=0,  # 横线的粗细，lw=0 时没有横线显示；
        label="Station")
]

ax.legend(
    handles=my_legend,
    prop=font,
    loc=2,
)

ax.set_xlabel(
    "Longitude",
    font="Times New Roman",
)
ax.set_ylabel(
    "Latitude",
    font="Times New Roman",
)
ax.set_title(
    "Station Geometry",
    size=20,
    font="Times New Roman",
)
plt.text(103.75,
         16.75,
         '10s',
         font=font,
         bbox=dict(
             boxstyle="round",
             ec=(1., 0.5, 0.5),
             fc=(1., 0.8, 0.8),
         ))
plt.savefig("../Figs/Raypath_Plot_10s.pdf")
plt.show()
