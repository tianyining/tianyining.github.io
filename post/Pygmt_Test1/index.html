说明

blockmean 模块计算不同块体内部的数量，其中块体尺寸大小是按照空间距离等距离划分的，下面的两个例子，分别展示了如何使用 blockmean 模块计算每个 block 内部地震事件的平均深度，以及统计每个 block 内部分别有多少个地震事件。

import os
import pygmt
import pandas as pd

data = pygmt.datasets.load_japan_quakes()
data.to_csv(
    "../InpFiles/japen_quakes.csv",
    index=None,
)
test_data = pd.read_csv("../InpFiles/japen_quakes.csv", sep=",")
test_data_part = data[["longitude", "latitude", "depth_km"]]
'''
 Set the region for plot 
 Define spacing in x and y direction (150 by 150 minute blocks)
'''
region = [130, 152.5, 32.5, 52.5]
spacing = "150m"
fig = pygmt.Figure()
'''
 Calculate mean depth in km from all events within each block 
 计算每个 block 中所有 events 的平均 depth 输入的数据 test_data_part.csv 就是每个地震经度、纬度、深度
 的信息汇总；
 生成的 df 文件包含三列：经度、纬度、平均深度；
'''
df = pygmt.blockmean(data=test_data_part, region=region, spacing=spacing)
df.to_csv("../InpFiles/df.csv")
'''
 Convert to grd File 
'''
grd = pygmt.xyz2grd(
    data=df,
    region=region,
    spacing=spacing,
)

fig.grdimage(
    grid=grd,
    region=region,
    frame=["af", '+t" Mean Earthquake Depth inside each Block"'],
    cmap="batlow",
)

fig.plot(x=test_data_part.longitude,
         y=test_data_part.latitude,
         style="c0.3c",
         color="white",
         pen="1p,black")

fig.coast(land="darkgrey", transparency=40)
fig.colorbar(frame = ['x+l"Average Events Depth(km)"'])  
fig.show()

import pygmt
import os
import pandas as pd

data = pd.read_csv("../InpFiles/japen_quakes.csv", sep=",")

test_data_part = data[["longitude", "latitude", "depth_km"]]
'''
 Set Region 
'''
fig = pygmt.Figure()
region = [130, 152.5, 32.5, 52.5]
spacing = "150m"
'''
 Calculate number of total locations within 150*150 minute bins via 
 blockmean's summary parameters 
'''
df = pygmt.blockmean(
    data=test_data_part, region=region, spacing=spacing,
    S="n")  # Use S="n" to report the number of input points inside each block,

grd = pygmt.xyz2grd(
    data=df,
    region=region,
    spacing=spacing,
)

fig.grdimage(
    grid=grd,
    region=region,
    frame=["af", '+t"Number of points inside each bloack"'],
    cmap="batlow",
)
fig.plot(
    x=test_data_part.longitude,
    y=test_data_part.latitude,
    style="c0.3c",
    pen="1p,black",
    color="white",
)
fig.coast(
    land="darkgrey",
    transparency=40,
)
fig.colorbar(frame=['x+l"Total Number of Events in each Blocks(count)"'])
fig.show()

import os
import pygmt
import pandas as pd
import numpy as np

data = pygmt.datasets.load_japan_quakes()
data.to_csv(
    "../InpFiles/japen_quakes.csv",
    index=None,
)
test_data = pd.read_csv("../InpFiles/japen_quakes.csv", sep=",")
test_data_part = data[["longitude", "latitude", "depth_km", "magnitude"]]
'''
 Set the region for plot 
'''
region = [130, 152.5, 32.5, 52.5]
spacing = "150m"
fig = pygmt.Figure()
fig.coast(
    region=region,
    land="darkgrey",
    #     transparency=40,
    frame=["af", '+t"Earthquake Distribution Map"'],
)
'''
 Define the colormap 
'''
pygmt.makecpt(
    cmap="viridis",
    revers=False,
    series=[0, 600, 100],
    output="../InpFiles/cmapmod.cpt",
    continuous=True,
)
'''
 Plot Earthquakes Distribution Map 
'''
fig.plot(
    x=test_data_part.longitude,
    y=test_data_part.latitude,
    style="c",
    size=np.exp(test_data_part.magnitude / 1.7) * 0.01,
    color=test_data_part.depth_km,
    pen="1p,black",
    cmap="../InpFiles/cmapmod.cpt",
)

fig.colorbar(
    cmap="../InpFiles/cmapmod.cpt",
    box=False,
    frame=['+l"Events Depth Variation(km)"'],
)
fig.show()
fig.savefig(
    "../Figs/Earthquakes Distribution Map.png",
    dpi=720,
)

import os
import pygmt
import pandas as pd
import numpy as np

data = pygmt.datasets.load_japan_quakes()
data.to_csv(
    "../InpFiles/japen_quakes.csv",
    index=None,
)
test_data = pd.read_csv("../InpFiles/japen_quakes.csv", sep=",")
test_data_part = test_data[["longitude", "latitude", "depth_km", "magnitude"]]
'''
 Set the region for plot 
'''
region = [130, 152.5, 32.5, 52.5]
spacing = "150m"
fig = pygmt.Figure()

df = pygmt.blockmean(data=test_data_part, region=region, spacing=spacing)
'''
 Convert to grd File 
'''
grd = pygmt.xyz2grd(
    data=df,
    region=region,
    spacing=spacing,
)
pygmt.makecpt(
    cmap="batlow",
    revers=False,
    series=[0, 600, 100],
    output="../InpFiles/batlow.cpt",
    continuous=True,
)
fig.grdimage(
    grid=grd,
    region=region,
    frame=["af", '+t" Mean Earthquake Depth inside each Block"'],
    cmap = "../InpFiles/batlow.cpt", 
)
fig.coast(
    land="darkgrey",
    transparency=40,
)
'''
 Define the Earthquakes Depth colormap
'''
pygmt.makecpt(
    cmap="viridis",
    revers=False,
    series=[0, 600, 100],
    output="../InpFiles/cmapmod.cpt",
    continuous=True,
)
'''
 Plot Earthquakes Distribution Map
'''
fig.plot(
    x=test_data_part.longitude,
    y=test_data_part.latitude,
    style="c",
    size=np.exp(test_data_part.magnitude / 1.7) * 0.01,
    color=test_data_part.depth_km,
    pen="1p,black",
    cmap="../InpFiles/cmapmod.cpt",
)

fig.colorbar(
    cmap="../InpFiles/cmapmod.cpt",
    box=False,
    frame=['+l"Events Depth Variation(km)"'],
)
# set fontsize of label to '12'
with pygmt.config(FONT_LABEL='12'):
    fig.colorbar(
        cmap="../InpFiles/batlow.cpt",
        position='x0.2c/15.7c+w6.5c/0.45c+ef0.2c+ml+h',
        box=False,
        frame=['+L"Average Depth in each Block"'], 
    )

col_leg = 'lightgray'

fig.shift_origin(xshift='10c', yshift='13.2c')

# plot circles
yval1 = 18 
fig.plot(region=[0, 26, 0, 26],
         projection='M4',
         x=22,
         y=yval1,
         style='c' + str(np.exp(7 / 1.7) * 0.01),
         pen='1.2p,black',
         color=col_leg)
fig.plot(x=14.3,
         y=yval1,
         style='c' + str(np.exp(6 / 1.7) * 0.01),
         pen='1.2p,black',
         color=col_leg)
fig.plot(x=9.3,
         y=yval1,
         style='c' + str(np.exp(5 / 1.7) * 0.01),
         pen='1.2p,black',
         color=col_leg)
fig.plot(x=5.6,
         y=yval1,
         style='c' + str(np.exp(4 / 1.7) * 0.01),
         pen='1.2p,black',
         color=col_leg)

# add annotations
yval2 = 12.5
fig.text(x=22, y=yval2, text='6', font='12p')
fig.text(x=14.3, y=yval2, text='5', font='12p')
fig.text(x=9.3, y=yval2, text='4', font='12p')
fig.text(x=5.6, y=yval2, text='3', font='12p')
fig.text(x=1.9, y=yval1, text='Mw', font='12p')


fig.savefig("../Figs/Blockmean_Test.png",dpi=720,)
fig.show()
