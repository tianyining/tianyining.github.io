QCstep1_Del_extremly_Bad

将存在极度异常值的相速度频散曲线，整支删除；具体来说，就是将频散曲线中存在相速度数值大于 5.5 以及小于 1 的整支频散曲线删掉。

import numpy as np
import glob
import matplotlib.pyplot as plt
import os
from pandas import DataFrame
from jupyterthemes import jtplot

jtplot.style(theme='monokai')

inDir = "/home/tianyining/Practice/MM_Whole/input/MM_DISP_Interp/"
output_dir = "/home/tianyining/Practice/MM_Whole/output/step1_del_extremely_Bad/"

os.chdir(inDir)
csvfiles = glob.glob("*.csv")
count_bad = 0
plt.figure(figsize=(12, 6.5))
for csvfile in csvfiles:
    data = np.loadtxt(csvfile, skiprows=1, delimiter=',')
    period = data[:, 0]
    phV = data[:, 1]
    snr = data[:, 2]
    if np.max(phV) > 5.5 or np.min(phV) < 1:
        sta1 = csvfile.split("_")[1]
        sta2 = csvfile.split("_")[2].split(".")[0]
        print(sta1, sta2, "skip")
        count_bad += 1
        continue

    period_array = np.array(period)[:, np.newaxis]
    phV_array = np.array(phV)[:, np.newaxis]
    snr_array = np.array(snr)[:, np.newaxis]
    concatenate_array = np.concatenate((period_array, phV_array, snr_array),
                                       axis=1)

    sgl_disp_file = DataFrame(concatenate_array,
                              columns=['period', 'ph_V', 'SNR'])
    sgl_disp_file.to_csv(output_dir + csvfile.split(".")[0] + "_QC1.csv",
                         index=None)
    # 设置坐标范围
#     plt.xlim((1, 55))
#     plt.ylim((0.8, 5.0))
    plt.title('PhV Dispersion Curves del Extremely Bad',
              fontsize='large', fontweight='normal')
    plt.plot(period, phV, 'r-')

print("bad SWD {}".format(count_bad / len(csvfiles)))
# plt.show()
plt.savefig(
    '/home/tianyining/Practice/MM_Whole/Figs/QCstep1_Del_Extremely_Bad.jpg')

QCstep2_del_snr

在 step1 中去掉存在异常值的整支频散曲线的基础上，对信噪比进行筛选，将 snr < 10 的相速度数值点删除，并存为新的文件；

这里只是将信噪比低的数据点删除，并非将相速度频散曲线整支删除；

'''
实现批量处理，将 SNR < 10 的数据点删除，另存为新的 CSV 文件；
'''
import pandas as pd
import glob
import os

input_dir = "/home/tianyining/Practice/MM_Whole/output/step1_del_extremely_Bad/"
output_dir = "/home/tianyining/Practice/MM_Whole/output/step2_del_snr/"

csvfiles = glob.glob(input_dir + "*")

for csvfile in csvfiles:
    sgl_file_name = os.path.basename(csvfile)
    sgl_file_new_name = sgl_file_name.split(".")[0] + "_snr" + ".csv"
    sgl_data_info = pd.read_csv(csvfile)
#     sgl_data_info["SNR"].describe()
    sgl_data_new = sgl_data_info[sgl_data_info["SNR"] > 10]
    sgl_data_new.to_csv(output_dir + sgl_file_new_name, index=None)
