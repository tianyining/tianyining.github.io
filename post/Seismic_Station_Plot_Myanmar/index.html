import pygmt
import os
import pandas as pd

# 读入台站信息
MM_stations = pd.read_csv('./station_mm.lst', sep='\s+')  # 正则表达式
'''
Generate Myanmar Map
'''
fig = pygmt.Figure()
# fig = pygmt.Figure.plot(size=(20, 20), dpi=100)
# fig = plt.figure(figsize=(16,8), dpi=100)
# 将 grdimage 信息导入到 KWARGS 中；
KWARGS = dict(
    grid='studyarea.grd',
    region=[91, 107, 16, 29],
    #     projection='M10i',
    projection='M10i',
    cmap='my2.cpt',
    frame=True,
    shading=False,
)

fig.grdimage(**KWARGS)

fig.coast(
    shorelines=True,
    #           borders = "1/0.8p",
    borders=["1/0.8p", "2/0.1p"],
    frame=True,
    map_scale='-68.5/7.0/7.0/200',
    resolution='f',
)

# 绘制台站分布；
fig.plot(
    x=MM_stations.sta_lon,
    y=MM_stations.sta_lat,
    style='i0.2i',
    color='purple',
    pen='black',
)

# 绘制断层线
fig.plot(data='./SF/SF.txt', pen="1.5,black")
fig.plot(data='./SF/SF_2.txt', pen="1.5,black")
fig.plot(data='./SF/SF_3.txt', pen="1.5,black")
fig.plot(data='./SF/SF_4.txt', pen="1.5,black")
fig.plot(data='./SF/SF_5.txt', pen="1.5,black")
fig.plot(data='./SF/SF_6.txt', pen="1.5,black")
fig.plot(data='./SF/SF_7.txt', pen="1.5,black")
fig.plot(data='./SF/SF_8.txt', pen="1.5,black")
fig.plot(data='./SF/SF_9.txt', pen="1.5,black")
fig.plot(data='./SF/SF_10.txt', pen="1.5,black")
fig.plot(data='./RedRiver/RRF.txt', pen="2.0,black")
# pygmt.Figure().plot(data='./SF/SF.txt', pen="1.5,black")
# fig.text()
fig.text(
    textfiles=None,
    x=96.3,
    y=21.7,
    position=None,
    text='Sagaing Fault',
    angle=-82,
    font='28p,Helvetica-Bold,black',
    justify='LM',
    #     fill = 'green',
)

fig.text(
    textfiles=None,
    x=103.0,
    y=22.3,
    position=None,
    text='RedRiver Fault',
    angle=-42,
    font='28p,Helvetica-Bold,black',
    justify='BL',
    #     fill = 'green',
)
'''
Pygmt Visualization 
'''
fig.show()
fig.savefig('./MM_test.pdf', crop=True, dpi=720)
