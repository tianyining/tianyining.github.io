import pandas as pd

estaciones = pd.read_csv('ESTACIONES.txt', sep='\s+')  # 正则表达式
estaciones = estaciones[estaciones.COD != "CARV"]  # 忽略 COD = 'CARV' 的第 7 行台站信息
nofun = pd.read_csv("ESTACIONES_NOFUNV.txt", sep='\s+')
'''
 Generate Map GMT : fig.grdimage
'''
fig = pygmt.Figure()
# 将 grdimage 信息存入 KWARGS 字典中
KWARGS = dict(
    grid='etopo1_bedrock.grd',
    region=[-76, -59, 5, 15],
    projection='M10i',
    cmap='verde.cpt',
    frame=True,
    shading=True,

fig.grdimage(**KWARGS)
'''
Generate Map GMT : fig.grdimage + fig.coast 
'''
fig.coast(
    shorelines=True,
    #           borders = "1/0.8p",

    frame=True,
    map_scale='-68.5/7.0/7.0/200',
    resolution='f',
)
'''
Generate Map GMT : fig.grdimage + fig.coast +  Geological faults
'''
fig.plot(data='fallas2015_test.txt', pen="1.2,red")
'''
Generate Map GMT : fig.grdimage + fig.coast + Geological faults + Stations
'''
# Plot national stations
fig.plot(
    x=estaciones.Lon,
    y=estaciones.Lat,
    style='i0.20i',
    color='yellow',
    pen='black',

# Plot international stations
fig.plot(
    x=nofun.Lon,
    y=nofun.Lat,
    style='t0.20i',  # t代表三角形表示，i 表示单位
    color='green',
    pen='black',
)
'''
 Generate Map GMT : fig.grdimage + fig.coast + Geological faults + Station text
'''

# Plot names of national stations(info from ESTACIONES.txt)
fig.text(
    textfiles=None,
    x=estaciones.Lon - 0.1,
    y=estaciones.Lat - 0.3,
    position=None,
    text=estaciones.COD,
    angle=0,
    font='10p,Helvetica-Bold,black',
    justify='LM',
)

# Plot names of international stations(info from ESTACIONES_NOFUNV.txt)
fig.text(
    textfiles=None,
    x=nofun.Lon - 0.1,
    y=nofun.Lat + 0.3,
    position=None,
    text=nofun.COD,
    angle=0,
    font='10p,Helvetica-Bold,black',
    justify='LM',
)

# Text for Caribbean Sea
fig.text(
    textfiles=None,
    x=-68.0,
    y=13.5,
    position=None,
    text='Caribbean Sea',
    angle=0,
    font='28p,Helvetica-Bold,white',
    justify='LM', 
    fill = 'green', 
)
'''
 Pygmt visualization
'''
fig.show()
fig.savefig('test_whole_text.png', crop=True, dpi=720)
