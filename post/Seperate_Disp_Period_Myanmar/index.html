output_dir = "../DISP_Period_Myanmar/"

import os
import glob
import numpy as np
import matplotlib.pyplot as plt
from jupyterthemes import jtplot

jtplot.style(theme="monokai")

plt.rcParams["axes.linewidth"] = 1  # 图框宽度；
plt.rcParams["figure.dpi"] = 720  #
plt.rcParams['font.sans-serif'] = ['Times New Roman']

input_dir = "../DISP_Period_Myanmar/"
output_dir = "../Figs/"

x = []
y = []
for sgl_disp in glob.glob(input_dir + "*"):
    sgl_disp_name = os.path.basename(sgl_disp)
    x.append(float(sgl_disp_name))

    tmpinfos = np.loadtxt(sgl_disp, delimiter="\t", usecols=(0, 1))

    GroupV = tmpinfos[:, 0]
    PhaseV = tmpinfos[:, 1]
    PhaseV_tuple = tuple(PhaseV)
    y.append(PhaseV_tuple)
# y = sorted(y)
'''

'''

        color="green",
        edgecolors="black",
    )
