'''
 Right Version 
'''
import os
import math
import numpy as np

input_dir = "../data_infos/"
output_dir = "../Output/"
data = np.loadtxt(input_dir + "disper_curves_compute")
shape = data.shape  # (193610, 2)
'''
establish dict
'''
dict_disp = {}
d = {}  # the value of d is standard deviation 标准差
a = {}  # the value of a is average 每个周期对应的相速度平均值；
s = {}  # sum velocity of each period 每个周期的相速度和
n = {}  # get the number of each period 每个周期对应有多少个相速度值
ss = {}  # sum of square(real phV - average)
'''
 已求取出： a-每个周期对应的相速度平均值；
          s-每个周期对应的相速度和;
          n-每个周期上分别有多少个相速度测量值；
'''
for i in range(shape[0]):
    period = int(data[i][0])
    if period not in s.keys():
        s[period] = data[i][1]
        n[period] = 1
    else:
        s[period] = s[period] + data[i][1]
        n[period] = n[period] + 1
    if i == shape[0] - 1:
        for item in s.keys():
            a[item] = s[item] / n[item]
'''
 index j 求取 ss-每个周期对应的平方和（测量值-周期平均值）的平方和 
            d-计算得到每个周期对应的标准差 
'''
for j in range(shape[0]):
    period = int(data[j][0])
    if period not in ss.keys():
        ss[period] = math.pow((data[j][1] - a[period]), 2)
    else:
        ss[period] = math.pow((data[j][1] - a[period]), 2) + ss[period]
    if j == shape[0] - 1:
        for item in ss.keys():
            d[item] = math.sqrt(ss[item] / n[item])
'''
 将计算结果写入新的频散文件，用于绘图和筛选
 each-period_average_std 三列分别代表：周期 - 周期对应的平均相速度 - 标准差
'''
with open(output_dir + "each-period_average_std", "w") as fp:
    for p in ss.keys():
        fp.write("{:<d} {:<.4f} {:<4f}\n".format(p, a[p], d[p]))

# 某些特定周期对应的相速度取值范围
with open(output_dir + "select-period_phvplus", "w") as fp:
    for item in range(5, 41, 5):
        print(item)
        fp.write("> \n")
        fp.write("{:<d} {:<.6f}\n".format(item, a[item] - 2 * d[item]))
        fp.write("{:<d} {:<.6f}\n".format(item, a[item] + 2 * d[item]))
'''
 计算自定义周期的射线路径
'''
t = 0
with open(output_dir + "number_ray_gmt", "w") as fp:
    for i in range(shape[0]):
        if data[i][1] < a[data[i][0]] - 2 * d[data[i][0]] or data[i][1] > a[
                data[i][0]] + 2 * d[data[i][0]]:
            n[data[i][0]] = n[data[i][0]] - 1
        else:
            t = t + 1
        if i == shape[0] - 1:
            for item in range(3, 41):
                fp.write("> \n")
                fp.write("{:<.1f} {:<d}\n".format(item - 0.5, n[item]))
                fp.write("{:<.1f} {:<d}\n".format(item + 0.5, n[item]))
                fp.write("{:<.1f} {:<d}\n".format(item + 0.5, 0))
                fp.write("{:<.1f} {:<d}\n".format(item - 0.5, 0))

输出的数据文件格式

each-period_average_std

生成的文件包含三列数据，周期、该周期对应的相速度平均值、std

注意：该文件周期与该周期对应的相速度平均值为键值对关系，因此不可能存在重复的周期。

select-period_phvplus（GMT 绘图文件）

挑选的特定周期对应的相速度取值范围文件（即GMT绘图的红线数据范围内的横纵坐标启始点）：

number_ray_gmt（GMT 绘图文件）

绘制不同周期射线路径的柱状统计分布图，每个 > > 中间表示的是一个柱子从左上角开始，顺时针方向计数，共 4 个点的坐标值。

该文件是 GMT 绘制柱状统计图的参考文件。

# Function: Plot number of ray paths in the period 
# Date:2021-11-05 
# Author:Yining Tian 

output="../Figs/dispersion_tyn"
inp="../data_infos/number_ray_gmt"

Rb="-R0/45/0/8000"
Jb="-JX4.5i/3.5i"
gmt begin ${output}_number_ray_path png 
    gmt basemap $Rb $Jb -Bya1000f500+l'number of ray paths' -Bxa5f2.5+l'period (s)' -BWS -X6i 
    gmt plot $inp $Rb $Jb -W0p -Ggray -L 
    echo "2 7600 (b) " |gmt pstext ${Rb} ${Jb} -F+f20p,8 
gmt end show
