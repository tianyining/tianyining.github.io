说明

本篇 blog 记录如何利用 AFTAN 从互相关对称分量中提取频散文件，计算不同周期的射线路径，并最终绘制绘制不同周期的台站对射线路径分布图；

Here we will calculate the paths. In this case we are simply assuming that the seismic waves propagate along great-circle paths between stations. This assumption may not be very good in areas with large velocity anomalies, but it works.

数据来源

本篇 blog 使用的数据来源如下：

  • PKU2020_Summer 中的 Demo
  • roses2021-unit03 中的 dataset

'''
对频散文件的观测周期进行线性插值；
首先读取频散文件中的观测周期，得到向上取整的观测周期的 int period list 再进行 1D Interpolate 
'''
import os
import glob
import numpy as np
from scipy import interpolate

input_disp = "../DISP/"
output_dir = "../DISP_Interp/"
fmt = "{:<f}\t{:<15.6f}\t{:<15.6f}\n"
for disp in glob.glob(input_disp + "*.SAC_s_2_DISP.1"):
    disp_name = os.path.basename(disp)

    fileID = open(disp, "r")
    tmp_info = fileID.readlines()
    fileID.close()

    observe_period = []
    Group_V = []
    Phase_V = []

    for i in range(0, len(tmp_info)):
        sgl_line = tmp_info[i].strip()
        period = float(sgl_line.split()[2])
        group_v = float(sgl_line.split()[3])
        phase_v = float(sgl_line.split()[4])

        observe_period.append(period)
        Group_V.append(group_v)
        Phase_V.append(phase_v)
    '''
    建立 int_period_list 
    '''
    period_list = []
    for item in observe_period[0:-1]:
        item_int = np.ceil(item)
        #     print(item_int)
        period_list.append(item_int)
    period_list.append(np.floor(observe_period[-1]))

    period_set = list(set(period_list))
    '''
    Interpolate Begin !!! 
    '''
    x = observe_period
    y = Group_V
    z = Phase_V

    # 定义插值函数 f
    f_groupv = interpolate.interp1d(x, y, kind="linear")
    f_phasev = interpolate.interp1d(x, z, kind="linear")

    x_new = period_set
    int_Group_V = f_groupv(x_new)
    int_Phase_V = f_phasev(x_new)
    '''
    将一维线性插值后的结果，重新保存为新的频散文件；
    '''
    with open("../DISP_Interp/new_" + disp_name, "w") as fileID:
        for n in range(0, len(x_new)):
            int_period = x_new[n]
            interp_groupv = int_Group_V[n]
            interp_phasev = int_Phase_V[n]

            fileID.write(fmt.format(int_period, interp_groupv, interp_phasev))

'''
首先读取对应台站对的 频散文件 & 互相关对称分量计算出的信噪比文件；
'''

import os
import glob
import pandas as pd
from scipy import interpolate

input_disp_dir = "../DISP_Interp/"
input_snr_dir = "../SNR_s/"
output_dir = "../DISP_SNR_Merge/"

for sgl_disp in glob.glob(input_disp_dir + "*.SAC_s_2_DISP.1"):
    sgl_disp_name = os.path.basename(sgl_disp)
    namescol_disp = ["int_period", "group_v", "phase_v"]
    dataset = pd.read_csv(
        input_disp_dir + sgl_disp,
        names=namescol_disp,
        delim_whitespace="\t",
    )

    int_period = dataset["int_period"]  # 作为 x_new
    group_v = dataset["group_v"]
    phase_v = dataset["phase_v"]
    '''
    读取对应的 snr_s.txt 文件；
    '''
    tmp = sgl_disp_name[4:].split(".")[0]
    tmp_snr = tmp + ".SAC_s_snr.txt"

    namescol_snr = ["period", "snr_s"]
    data_snr = pd.read_csv(
        input_snr_dir + tmp_snr,
        names=namescol_snr,
        delim_whitespace=True,
    )
    period_snr = data_snr["period"]  # 作为 x
    snr_s = data_snr["snr_s"]  # 作为 y
    '''
    interpolate begin !!! 
    '''
    x = period_snr
    y = snr_s

    # 定义插值函数 f
    f_snr = interpolate.interp1d(x, y, kind="linear")

    x_new = int_period
    interp_snr = f_snr(x_new)
    '''
    将 snr_s 一维线性插值后的结果与 DISP_Interp 结合在一起，生成新的频散文件；
    新的频散文件包含四列信息： int_period group_v phase_v snr_s 
    '''
    with open(output_dir + sgl_disp_name, "w") as fileID:
        for n in range(0, len(int_period)):
            p = int_period[n]
            G_v = group_v[n]
            Pha_v = phase_v[n]
            SNR_s = interp_snr[n]

            fileID.write("{:<.1f}\t{:<15.6f}\t{:<15.6f}\t{:<15.6f}\n".format(
                p, G_v, Pha_v, SNR_s))

'''
 将频散文件整理成以整数周期命名的格式，其中每个周期包含 10 列信息：
 sta1 sta2 groupv phasev snr sta1_lon sta1_lat sta2_lon sta2_lat dist 
 注意：该脚本只可执行一次，重复执行需要将 output_dir 中的频散文件清空 ！！！！ 
'''
import os
import glob
import pandas as pd
import numpy as np

input_disp_dir = "../DISP_SNR_Merge/"
output_dir = "../DISP_Period/"

for disp_file in glob.glob(input_disp_dir + "*.SAC_s_2_DISP.1"):
    disp_file_name = os.path.basename(disp_file)
    namescol = ["int_period", "Group_V", "Phase_V", "SNR_s"]
    disp_dataset = pd.read_csv(
        disp_file,
        names=namescol,
        delim_whitespace=True,
    )

    sta1, sta2 = disp_file_name.split("_")[2], disp_file_name.split(
        "_")[3].split(".")[0]

    print(sta1, sta2)

    # 读取每个频散文件的内容；
    fileID = open(disp_file, "r")
    tmpinfo = fileID.readlines()
    fileID.close()

    Period, GroupV, PhaseV, SNR_s = [], [], [], []

    for line in tmpinfo:
        line = line.strip("\n").split()
        PER = float(line[0])
        G_V = float(line[1])
        Pha_V = float(line[2])
        SNR = float(line[3])
        Period.append(PER)
        GroupV.append(G_V)
        PhaseV.append(Pha_V)
        SNR_s.append(SNR)

    PER, GRP, PHS, snr = [], [], [], [
    ]  # PER GRP PHS snr same with Period GroupV PhaseV SNR_s
    for nn in range(len(Period)):
        tmpper = Period[nn]

        PER.append(tmpper)
        GRP.append(GroupV[nn])
        PHS.append(PhaseV[nn])
        snr.append(SNR_s[nn])
    '''
    Write into new disp files 
    '''
    for i in range(len(PER)):
        index = np.where(np.array(Period) == PER[nn])[0]  # 返回索引；

        if list(index) == []:
            continue
        else:
            index = index[0]

        fileID = open(output_dir + str(int(PER[i])) + ".dat", "a")
        fileID.write(
            "{:<}\t{:<}\t{:<.6f}\t{:<6f}\t{:<.6f}\t{:<.6f}  {:<.6f}  {:<.6f}  {:<.6f}\t{:<.6f} \n"
            .format(
                sta1,
                sta2,
                GRP[i],
                PHS[i],
                snr[i],
                Sta_Info[sta1]["sta_lon"],
                Sta_Info[sta1]["sta_lat"],
                Sta_Info[sta2]["sta_lon"],
                Sta_Info[sta2]["sta_lat"],
                CCF_INFO[disp_file_name[4:][0:-11]]['ccf_dist'],
            ))

import pyproj

wgs84 = pyproj.Geod(ellps='WGS84')
def dist(lons1, lats1, lons2, lats2):
    """
    Returns an array of geodetic distance(s) in km between
    points (lon1, lat1) and (lon2, lat2)
    """
    _, _, d = wgs84.inv(lons1=lons1, lats1=lats1, lons2=lons2, lats2=lats2)
    return np.array(d) / 1000.0

def geodesic(coord1, coord2, npts):
    """
    Returns a list of *npts* points along the geodesic between
    (and including) *coord1* and *coord2*, in an array of
    shape (*npts*, 2).
    @rtype: L{ndarray}
    """
    if npts < 2:
        raise Exception('nb of points must be at least 2')

    path = wgs84.npts(lon1=coord1[0], lat1=coord1[1],
                      lon2=coord2[0], lat2=coord2[1],
                      npts=npts-2)
    return np.array([coord1] + path + [coord2])

'''
Modified from seismic-noise-tomography 
Calculate Ray Paths 
'''
paths = []  # 10s 周期，测量不同台站对间的频散，有多少个台站对，就有多少条 path
dists = []  # 10s 周期，测量不同台站对间的 dist_test 有多少个台站对，就有多少个 dist_test
coords = []

path_num = 0
for index, row in disp_curves_clean.iterrows():

    # calculate the distance between two points(two stations)
    dist_test = dist(row.source_lon, row.source_lat, row.receiver_lon,
                     row.receiver_lat)
    '''
    npts 当 dist_test > 100 时，为 dist_test 本身，否则为 100 
    '''
    npts = np.max([np.ceil(dist_test) + 1, 100])

    source_coords = (row.source_lon, row.source_lat)
    if source_coords not in coords:
        coords.append(source_coords)
    receiver_coords = (row.receiver_lon, row.receiver_lat)
    if receiver_coords not in coords:
        coords.append(receiver_coords)
    '''
    Calculate the path between source_coords & receiver_coords 
    '''
    path = geodesic(source_coords, receiver_coords, npts)
    path_num += 1
    paths.append(path)
    dists.append(dist_test)
'''
 number of dists : 323 
 path_num = 323 
 
'''
dists = np.array(dists)
paths = np.array(paths, dtype="object")
