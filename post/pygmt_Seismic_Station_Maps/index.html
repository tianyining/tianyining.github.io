import pygmt
from jupyterthemes import jtplot

jtplot.style(theme='monokai')
minlon, maxlon = 91, 105
minlat, maxlat = 16, 29

# Define etopo data file
topo_data = '@earth_relief_15s'

# Initialize the Pygmt figure
fig = pygmt.Figure()

# Define CPT file:make color pallets
pygmt.makecpt(
    cmap='terra',
    series='-8000/8000/1000',
    continuous=True,
)

# Plot high resolution tomography from data source
fig.grdimage(
    grid=topo_data,
    region=[minlon, maxlon, minlat, maxlat],
    projection='M4i',
    frame=True,  # 边框
    #     shading = True,
)

# Plot the coastlines/shorelines on the map 绘制海岸线
fig.coast(
    region=[minlon, maxlon, minlat, maxlat],
    projection='M4i',
    shorelines=True,
    frame=True,
)

# Plot the topographic contout lines
fig.grdcontour(
    grid=topo_data,
    interval=4000,
    annotation='4000+f6p',
    #     limit="-8000/0",  # only show contours with elevation less than 0 km;
    limit="-8000/8000",
    pen="a0.15p")

fig.plot(sta_lons, sta_lats, style='i0.15i', color='blue', pen=True)
fig.savefig('station_map_MMtest.png', crop=True, dpi=720)
fig.show()
