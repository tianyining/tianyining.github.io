'''
 使用 pandas 读入台站信息；
'''

Sta_File = "./Infos_tyn/sta_coordinates.txt"
namescol = ["staname", "stalon", "stalat"]
dataset = pd.read_table(Sta_File, delim_whitespace=True, names=namescol)

'''
 Mark Stations that should be Highlight !!! 
 Only plot blue circle 
 只能执行一次，否则 Highlight 部分无法出渐变效果；
'''
Sta_Highlight_List_blue = ["KEV", "HAMF", "VAF", "NWG28", "SA64", "SA39"]
circsize = "c0.35c"  # 定义圆圈的大小；
circcol = "blue"  # 定义圆圈的填充颜色；
circthick = "0.034c"  # 定义圆圈线条厚度；
circtrans = "50"  # "blue@50" 产生渐变的色彩；
a = dataset.stalon[dataset["staname"].isin(Sta_Highlight_List_blue)]
b = dataset.stalat[dataset["staname"].isin(Sta_Highlight_List_blue)]

fig.plot(x=dataset.stalon[dataset["staname"].isin(Sta_Highlight_List_blue)],
         y=dataset.stalat[dataset["staname"].isin(Sta_Highlight_List_blue)],
         style=circsize,
         pen=circthick + "," + circcol,
         color=circcol + '@' + circtrans)

fig.show()
fig.savefig("./Figs_test/Fig_Sta_Highlight.png", dpi=720)

'''
 通过定义函数的方式，绘制剩余不同来源的台站，不同来源用不同颜色表示；
'''
stasym = "t0.16c"
staedg = "0.015c" + ",black"


# helper function
def plot_stas(staid, stacol, stalab):
    fig.plot(
        x=dataset.stalon[dataset['staname'].str.startswith(staid)],
        y=dataset.stalat[dataset['staname'].str.startswith(staid)],
        style=stasym,
        pen=staedg,
        color=stacol,
        label=stalab,
    )


plot_stas("SA", "yellow", '"ScanArray Core"')  # 用黄色绘制 "SA" 开头的台站
plot_stas(("N2", "NBB"), 'orange', "Neonor2")  # 以橙色绘制 "N2" "NBB" 开头的台站
plot_stas('N1', 'dodgerblue', "SCANLIPS3D")  # 以 dodgerblue 绘制 "N1" 开头的台站
plot_stas("NWG", '0/205/102', "MAGNUS")  # 以 绿色 绘制 "NWG" 开头的台站

fig.show()
fig.savefig("./Figs_test/Fig_Sta_Whole.png", dpi=720)

                  FONT_LABEL='6',
                  MAP_FRAME_PEN='0.4p'):
    fig.legend(
        position='JBR+jBR+o0.1c',  # 放在右下角，偏移 0.1cm 
        box='+plinen+glinen+gwhite+p0.5p,black')
    # which draws a box around the legend using a 1p black pen and adds a white background.

        MAP_TICK_LENGTH_PRIMARY='0.05',

    fig.colorbar(cmap = 'topo.cpt',
                 position="x0.1i/2.085i+w1.2i/0.075i",
                 frame=['x1000+l"topography in m"'],
                 box = '+r2p+pwhite+gwhite@20+p0.5p')
