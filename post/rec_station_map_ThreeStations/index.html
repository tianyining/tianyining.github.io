import pandas as pd
import os
import matplotlib.pyplot as plt
from jupyterthemes import jtplot

jtplot.style(theme="monokai")
DATADIR = '../data/'
station_meta = pd.read_csv(DATADIR + '/meta.csv')
station_meta = pd.read_csv(os.path.join(DATADIR, 'meta.csv'))

rec_sta1 = 'M07A'
rec_sta2 = 'M15A'

# 提取 rec_sta1 rec_sta2 两个接收台站的经纬度信息 lon1 lat1; lon2 lat2
'''
根据接收台站名，提取两个接收台站的经纬度信息，
根据 staN_info.shape 选择 staN_info.iloc[x,y] 的坐标值；
'''

sta1_info = station_meta[station_meta['sta'] == rec_sta1]
sta2_info = station_meta[station_meta['sta'] == rec_sta2]
sta1_lon = sta1_info.iloc[0, 2]
# print(sta1_lon)
sta1_lat = sta1_info.iloc[0, 3]
# print(sta1_lat)
sta2_lon = sta2_info.iloc[0, 2]
# print(sta2_lon)
sta2_lat = sta2_info.iloc[0, 3]
# print(sta2_lat)
'''
 建立图幅，并绘制台站分布图
  1. 首先将除接收台站外的所有源台站绘制出，用 'o' 圆圈表示；
  2. 再将其中的两个接收台站： rec_sta1 rec_sta2 单独绘制出来，用 '^' 三角形表示；
'''
fig, ax = plt.subplots(1, 1, figsize=(12, 10), dpi=100)

ax.scatter(station_meta_new['lon'],
           station_meta_new['lat'],
           s=10,
           marker='o',
           color='red')
'''
 单独绘制 rec_sta1 rec_sta2
'''
ax.scatter(sta1_lon, sta1_lat, s=38, marker='^', color='blue')
ax.scatter(sta2_lon, sta2_lat, s=38, marker='^', color='blue')
'''
 Visualization 
'''

plt.savefig("../data/figs/Three_Station_Fig.png")
